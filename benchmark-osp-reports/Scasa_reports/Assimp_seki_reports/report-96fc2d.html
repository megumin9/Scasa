539aiMesh *X3DGeoHelper::make_mesh(const std::vector<int32_t> &pCoordIdx, const std::list<aiVector3D> &pVertices) {
540 std::vector<aiFace> faces;
541 unsigned int prim_type = 0;
542
543 // create faces array from input string with vertices indices.
544 X3DGeoHelper::coordIdx_str2faces_arr(pCoordIdx, faces, prim_type);
545 if (!faces.size()) {
546 throw DeadlyImportError("Failed to create mesh, faces list is empty.");
547 }
548
549 //
550 // Create new mesh and copy geometry data.
551 //
552 aiMesh *tmesh = new aiMesh;
553 size_t ts = faces.size();
554 // faces
555 tmesh->mFaces = new aiFace[ts];
556 tmesh->mNumFaces = static_cast<unsigned int>(ts);
557 for (size_t i = 0; i < ts; i++)
558 tmesh->mFaces[i] = faces.at(i);
559
560 // vertices

562
563 ts = pVertices.size();
564 tmesh->mVertices = new aiVector3D[ts];
565 tmesh->mNumVertices = static_cast<unsigned int>(ts);
566 for (size_t i = 0; i < ts; i++) {
567 tmesh->mVertices[i] = *vit++;
568 }
569
570 // set primitives type and return result.
571 tmesh->mPrimitiveTypes = prim_type;
572
573 return tmesh;
