471
472 const aiMesh *pcMesh = pcScene->mMeshes[i];
473 const ASE::Mesh *mesh = (const ASE::Mesh *)pcMesh->mColors[2];
474 if (mesh == snode) {
475 node->mMeshes[p++] = i;
476
477 // Transform all vertices of the mesh back into their local space ->
478 // at the moment they are pretransformed
479 aiMatrix4x4 m = mesh->mTransform;
480 m.Inverse();
481

513 ai_assert(4 <= AI_MAX_NUMBER_OF_COLOR_SETS)(void)((!!(4 <= 0x8)) || (Assimp::aiAssertViolation("4 <= AI_MAX_NUMBER_OF_COLOR_SETS"
, "/usr/STLcheck/sekitest/assimp/code/AssetLib/ASE/ASELoader.cpp"
, 513), 0));
514
515 // Receives child nodes for the pcParent node
516 std::vector<aiNode *> apcNodes;
517
518 // Now iterate through all nodes in the scene and search for one
519 // which has *us* as parent.
520 for (std::vector<BaseNode *>::const_iterator it = nodes.begin(), end = nodes.end(); it != end; ++it) {
521 const BaseNode *snode = *it;
522 if (szName) {
523 if (len != snode->mParent.length() || ::strcmp(szName, snode->mParent.c_str()))
524 continue;
525 } else if (snode->mParent.length())
526 continue;
527
528 (*it)->mProcessed = true;
529
530 // Allocate a new node and add it to the output data structure
531 apcNodes.push_back(new aiNode());
532 aiNode *node = apcNodes.back();
533
534 node->mName.Set((snode->mName.length() ? snode->mName.c_str() : "Unnamed_Node"));
535 node->mParent = pcParent;
536
537 // Setup the transformation matrix of the node
538 aiMatrix4x4 mParentAdjust = mat;
539 mParentAdjust.Inverse();
540 node->mTransformation = mParentAdjust * snode->mTransform;
541
542 // Add sub nodes - prevent stack overflow due to recursive parenting
543 if (node->mName != node->mParent->mName && node->mName != node->mParent->mParent->mName) {
544 AddNodes(nodes, node, node->mName.data, snode->mTransform);
545 }
546
547 // Further processing depends on the type of the node
548 if (snode->mType == ASE::BaseNode::Mesh) {
549 // If the type of this node is "Mesh" we need to search
550 // the list of output meshes in the data structure for
551 // all those that belonged to this node once. This is
552 // slightly inconvinient here and a better solution should
553 // be used when this code is refactored next.
554 AddMeshes(snode, node);
555 } else if (is_not_qnan(snode->mTargetPosition.x)) {
556 // If this is a target camera or light we generate a small
557 // child node which marks the position of the camera
558 // target (the direction information is contained in *this*
559 // node's animation track but the exact target position
560 // would be lost otherwise)
561 if (!node->mNumChildren) {
562 node->mChildren = new aiNode *[1];
563 }
564
565 aiNode *nd = new aiNode();
566
567 nd->mName.Set(snode->mName + ".Target");
568
569 nd->mTransformation.a4 = snode->mTargetPosition.x - snode->mTransform.a4;
570 nd->mTransformation.b4 = snode->mTargetPosition.y - snode->mTransform.b4;
571 nd->mTransformation.c4 = snode->mTargetPosition.z - snode->mTransform.c4;
572
573 nd->mParent = node;
574
575 // The .Target node is always the first child node
576 for (unsigned int m = 0; m < node->mNumChildren; ++m)
577 node->mChildren[m + 1] = node->mChildren[m];
578
579 node->mChildren[0] = nd;
580 node->mNumChildren++;
581
582 // What we did is so great, it is at least worth a debug message
583 ASSIMP_LOG_VERBOSE_DEBUG("ASE: Generating separate target node (", snode->mName, ")")Assimp::DefaultLogger::get()->verboseDebug("ASE: Generating separate target node ("
, snode->mName, ")");
584 }
585 }
586
587 // Allocate enough space for the child nodes
588 // We allocate one slot more in case this is a target camera/light
589 pcParent->mNumChildren = (unsigned int)apcNodes.size();
590 if (pcParent->mNumChildren) {
591 pcParent->mChildren = new aiNode *[apcNodes.size() + 1 /* PLUS ONE !!! */];
592
593 // now build all nodes for our nice new children
594 for (unsigned int p = 0; p < apcNodes.size(); ++p)
595 pcParent->mChildren[p] = apcNodes[p];
596 }
597 return;
598}

600// ------------------------------------------------------------------------------------------------
601// Build the output node graph
602void ASEImporter::BuildNodes(std::vector<BaseNode *> &nodes) {
603 ai_assert(nullptr != pcScene)(void)((!!(nullptr != pcScene)) || (Assimp::aiAssertViolation
("nullptr != pcScene", "/usr/STLcheck/sekitest/assimp/code/AssetLib/ASE/ASELoader.cpp"
, 603), 0));
604
605 // allocate the one and only root node
606 aiNode *root = pcScene->mRootNode = new aiNode();
607 root->mName.Set("<ASERoot>");
608
609 // Setup the coordinate system transformation
610 pcScene->mRootNode->mNumChildren = 1;
611 pcScene->mRootNode->mChildren = new aiNode *[1];
612 aiNode *ch = pcScene->mRootNode->mChildren[0] = new aiNode();
613 ch->mParent = root;
614
615 // Change the transformation matrix of all nodes
616 for (BaseNode *node : nodes) {
617 aiMatrix4x4 &m = node->mTransform;
618 m.Transpose(); // row-order vs column-order
619 }
620
621 // add all nodes
622 AddNodes(nodes, ch, nullptr);
623
624 // now iterate through al nodes and find those that have not yet
625 // been added to the nodegraph (= their parent could not be recognized)
626 std::vector<const BaseNode *> aiList;
627 for (std::vector<BaseNode *>::iterator it = nodes.begin(), end = nodes.end(); it != end; ++it) {
628 if ((*it)->mProcessed) {
629 continue;
630 }
631
632 // check whether our parent is known
633 bool bKnowParent = false;
634
635 // search the list another time, starting *here* and try to find out whether
636 // there is a node that references *us* as a parent
637 for (std::vector<BaseNode *>::const_iterator it2 = nodes.begin(); it2 != end; ++it2) {
638 if (it2 == it) {
639 continue;
640 }
641
642 if ((*it2)->mName == (*it)->mParent) {
643 bKnowParent = true;
644 break;
645 }
646 }
647 if (!bKnowParent) {
648 aiList.push_back(*it);
649 }
650 }
651
652 // Are there any orphaned nodes?
653 if (!aiList.empty()) {
654 std::vector<aiNode *> apcNodes;
655 apcNodes.reserve(aiList.size() + pcScene->mRootNode->mNumChildren);
656
657 for (unsigned int i = 0; i < pcScene->mRootNode->mNumChildren; ++i)
658 apcNodes.push_back(pcScene->mRootNode->mChildren[i]);
659
660 delete[] pcScene->mRootNode->mChildren;
661 for (std::vector<const BaseNode *>::/*const_*/ iterator i = aiList.begin(); i != aiList.end(); ++i) {
662 const ASE::BaseNode *src = *i;
663
664 // The parent is not known, so we can assume that we must add
665 // this node to the root node of the whole scene
666 aiNode *pcNode = new aiNode();
667 pcNode->mParent = pcScene->mRootNode;
668 pcNode->mName.Set(src->mName);
669 AddMeshes(src, pcNode);
670 AddNodes(nodes, pcNode, pcNode->mName.data);
671 apcNodes.push_back(pcNode);
672 }
673
674 // Regenerate our output array
675 pcScene->mRootNode->mChildren = new aiNode *[apcNodes.size()];
676 for (unsigned int i = 0; i < apcNodes.size(); ++i)
677 pcScene->mRootNode->mChildren[i] = apcNodes[i];
678
679 pcScene->mRootNode->mNumChildren = (unsigned int)apcNodes.size();
680 }
681
682 // Reset the third color set to nullptr - we used this field to store a temporary pointer
683 for (unsigned int i = 0; i < pcScene->mNumMeshes; ++i)
684 pcScene->mMeshes[i]->mColors[2] = nullptr;
685
686 // The root node should not have at least one child or the file is valid
687 if (!pcScene->mRootNode->mNumChildren) {
688 throw DeadlyImportError("ASE: No nodes loaded. The file is either empty or corrupt");
689 }
690
691 // Now rotate the whole scene 90 degrees around the x axis to convert to internal coordinate system
692 pcScene->mRootNode->mTransformation = aiMatrix4x4(1.f, 0.f, 0.f, 0.f,
693 0.f, 0.f, 1.f, 0.f, 0.f, -1.f, 0.f, 0.f, 0.f, 0.f, 0.f, 1.f);
694}

696// ------------------------------------------------------------------------------------------------
697// Convert the imported data to the internal verbose representation
698void ASEImporter::BuildUniqueRepresentation(ASE::Mesh &mesh) {
699 // allocate output storage
700 std::vector<aiVector3D> mPositions;
701 std::vector<aiVector3D> amTexCoords[AI_MAX_NUMBER_OF_TEXTURECOORDS0x8];
702 std::vector<aiColor4D> mVertexColors;
703 std::vector<aiVector3D> mNormals;
704 std::vector<BoneVertex> mBoneVertices;
705
706 unsigned int iSize = (unsigned int)mesh.mFaces.size() * 3;
707 mPositions.resize(iSize);
708
709 // optional texture coordinates
710 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_TEXTURECOORDS0x8; ++i) {
711 if (!mesh.amTexCoords[i].empty()) {
712 amTexCoords[i].resize(iSize);
713 }
714 }
715 // optional vertex colors
716 if (!mesh.mVertexColors.empty()) {
717 mVertexColors.resize(iSize);
718 }
719
720 // optional vertex normals (vertex normals can simply be copied)
721 if (!mesh.mNormals.empty()) {
722 mNormals.resize(iSize);
723 }
724 // bone vertices. There is no need to change the bone list
725 if (!mesh.mBoneVertices.empty()) {
726 mBoneVertices.resize(iSize);
727 }
728
729 // iterate through all faces in the mesh
730 unsigned int iCurrent = 0, fi = 0;
731 for (std::vector<ASE::Face>::iterator i = mesh.mFaces.begin(); i != mesh.mFaces.end(); ++i, ++fi) {
732 for (unsigned int n = 0; n < 3; ++n, ++iCurrent) {
733 mPositions[iCurrent] = mesh.mPositions[(*i).mIndices[n]];
734
735 // add texture coordinates
736 for (unsigned int c = 0; c < AI_MAX_NUMBER_OF_TEXTURECOORDS0x8; ++c) {
737 if (mesh.amTexCoords[c].empty()) break;
738 amTexCoords[c][iCurrent] = mesh.amTexCoords[c][(*i).amUVIndices[c][n]];
739 }
740 // add vertex colors
741 if (!mesh.mVertexColors.empty()) {
742 mVertexColors[iCurrent] = mesh.mVertexColors[(*i).mColorIndices[n]];
743 }
744 // add normal vectors
745 if (!mesh.mNormals.empty()) {
746 mNormals[iCurrent] = mesh.mNormals[fi * 3 + n];
747 mNormals[iCurrent].Normalize();
748 }
749
750 // handle bone vertices
751 if ((*i).mIndices[n] < mesh.mBoneVertices.size()) {
752 // (sometimes this will cause bone verts to be duplicated
753 // however, I' quite sure Schrompf' JoinVerticesStep
754 // will fix that again ...)
755 mBoneVertices[iCurrent] = mesh.mBoneVertices[(*i).mIndices[n]];
756 }
757 (*i).mIndices[n] = iCurrent;
758 }
759 }
760
761 // replace the old arrays
762 mesh.mNormals = mNormals;
763 mesh.mPositions = mPositions;
764 mesh.mVertexColors = mVertexColors;
765
766 for (unsigned int c = 0; c < AI_MAX_NUMBER_OF_TEXTURECOORDS0x8; ++c)
767 mesh.amTexCoords[c] = amTexCoords[c];
768}

976 if (!mesh.mBones.empty()) {

, "/usr/STLcheck/sekitest/assimp/code/AssetLib/ASE/ASELoader.cpp"

983 blubb != mesh.mBoneVertices[iIndex2].mBoneWeights.end(); ++blubb) {
984
985 // NOTE: illegal cases have already been filtered out
986 avOutputBones[(*blubb).first].emplace_back(
987 iBase, (*blubb).second);
988 }
989 }
990 }
991 p_pcOut->mFaces[q].mIndices[t] = iBase;
992 }
993 }
994 }
995 // convert texture coordinates (up to AI_MAX_NUMBER_OF_TEXTURECOORDS sets supported)
996 for (unsigned int c = 0; c < AI_MAX_NUMBER_OF_TEXTURECOORDS0x8; ++c) {
997 if (!mesh.amTexCoords[c].empty()) {
998 p_pcOut->mTextureCoords[c] = new aiVector3D[p_pcOut->mNumVertices];

1009 }
1010

1012 if (!mesh.mVertexColors.empty()) {

1015 for (unsigned int q = 0; q < aiSplit[p].size(); ++q) {
1016 iIndex = aiSplit[p][q];
1017 for (unsigned int t = 0; t < 3; ++t) {
1018 p_pcOut->mColors[0][iBase++] = mesh.mVertexColors[mesh.mFaces[iIndex].mIndices[t]];
1019 }
1020 }
1021 }
1022 // Copy bones
1023 if (!mesh.mBones.empty()) {
1024 p_pcOut->mNumBones = 0;
1025 for (unsigned int mrspock = 0; mrspock < mesh.mBones.size(); ++mrspock)
1026 if (!avOutputBones[mrspock].empty()) p_pcOut->mNumBones++;
1027
1028 p_pcOut->mBones = new aiBone *[p_pcOut->mNumBones];

1030 for (unsigned int mrspock = 0; mrspock < mesh.mBones.size(); ++mrspock) {
1031 if (!avOutputBones[mrspock].empty()) {
1032 // we will need this bone. add it to the output mesh and
1033 // add all per-vertex weights
1034 aiBone *pc = *pcBone = new aiBone();
1035 pc->mName.Set(mesh.mBones[mrspock].mName);
1036
1037 pc->mNumWeights = (unsigned int)avOutputBones[mrspock].size();
1038 pc->mWeights = new aiVertexWeight[pc->mNumWeights];
1039
1040 for (unsigned int captainkirk = 0; captainkirk < pc->mNumWeights; ++captainkirk) {
1041 const std::pair<unsigned int, float> &ref = avOutputBones[mrspock][captainkirk];
1042 pc->mWeights[captainkirk].mVertexId = ref.first;
1043 pc->mWeights[captainkirk].mWeight = ref.second;
1044 }
1045 ++pcBone;
1046 }
1047 }
1048 // delete allocated storage
1049 delete[] avOutputBones;
1050 }
1051 }
1052 }
1053 // delete storage
1054 delete[] aiSplit;
1055 } else {
