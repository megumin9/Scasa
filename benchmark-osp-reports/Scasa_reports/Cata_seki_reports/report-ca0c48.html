671static bool spell_infinite_loop_check( std::set<spell_id> spell_effects, const spell_id &sp )
672{
673 if( spell_effects.count( sp ) ) {
674 return true;
675 }
676 spell_effects.emplace( sp );
677
678 std::set<spell_id> unique_spell_list;
679 for( const fake_spell &fake_sp : sp->additional_spells ) {
680 unique_spell_list.emplace( fake_sp.id );
681 }
682
683 for( const spell_id &other_sp : unique_spell_list ) {
684 if( spell_infinite_loop_check( spell_effects, other_sp ) ) {
685 return true;
686 }
687 }
688 return false;
689}

1338float spell::spell_fail( const Character &guy ) const
1339{
1340 if( has_flag( spell_flag::NO_FAIL ) ) {
1341 return 0.0f;
1342 }
1343 const bool is_psi = has_flag( spell_flag::PSIONIC );
1344
1345 // formula is based on the following:
1346 // exponential curve
1347 // effective skill of 0 or less is 100% failure
1348 // effective skill of 8 (8 int, 0 spellcraft, 0 spell level, spell difficulty 0) is ~50% failure
1349 // effective skill of 30 is 0% failure
1350 const float effective_skill = 2 * ( get_effective_level() - get_difficulty(
1351 guy ) ) + guy.get_int() +
1352 guy.get_skill_level( skill() );
1353
1354 // skill for psi powers downplays power level and is much more based on level and intelligence
1355 // and goes up to 40 max--effective skill of 10 is 50% failure, effective skill of 40 is 0%
1356 // Int 8, Metaphysics 2, level 1, difficulty 1 is effective level 26.5
1357 // Int 10, Metaphysics 5, level 4, difficulty 5 is effective level 27
1358 // Int 12, Metaphysics 8, level 7, difficulty 10 is effective level 33.5
1359 const float two_thirds_power_level = static_cast<float>( get_effective_level() ) /
1360 static_cast<float>
1361 ( 1.5 );
1362
1363 const float psi_effective_skill = 2 * ( ( guy.get_skill_level( skill() ) * 2 ) - get_difficulty(
1364 guy ) ) + ( guy.get_int() * 1.5 ) + two_thirds_power_level;
1365 // add an if statement in here because sufficiently large numbers will definitely overflow because of exponents
1366 if( ( effective_skill > 30.0f && !is_psi ) || ( psi_effective_skill > 40.0f && is_psi ) ) {
1367 return 0.0f;
1368 } else if( ( effective_skill < 0.0f && !is_psi ) || ( psi_effective_skill < 0.0f && is_psi ) ) {
1369 return 1.0f;
1370 }
1371
1372 float fail_chance = std::pow( ( effective_skill - 30.0f ) / 30.0f, 2 );
1373 float psi_fail_chance = std::pow( ( psi_effective_skill - 40.0f ) / 40.0f, 2 );
1374
1375 if( !is_psi ) {
1376 if( has_flag( spell_flag::SOMATIC ) &&
1377 !guy.has_flag( json_flag_SUBTLE_SPELL ) && temp_somatic_difficulty_multiplyer > 0 ) {
1378 // the first 20 points of encumbrance combined is ignored
1379 const int arms_encumb = std::max( 0,
1380 guy.avg_encumb_of_limb_type( body_part_type::type::arm ) - 10 );
1381 // each encumbrance point beyond the "gray" color counts as half an additional fail %
1382 fail_chance += ( arms_encumb / 200.0f ) * temp_somatic_difficulty_multiplyer;
1383 }
1384 if( has_flag( spell_flag::VERBAL ) &&
1385 !guy.has_flag( json_flag_SILENT_SPELL ) && temp_sound_multiplyer > 0 ) {
1386 // a little bit of mouth encumbrance is allowed, but not much
1387 const int mouth_encumb = std::max( 0,
1388 guy.avg_encumb_of_limb_type( body_part_type::type::mouth ) - 5 );
1389 fail_chance += ( mouth_encumb / 100.0f ) * temp_sound_multiplyer;
1390 }
1391 }
1392
1393 // concentration spells work better than you'd expect with a higher focus pool
1394 if( has_flag( spell_flag::CONCENTRATE ) && temp_concentration_difficulty_multiplyer > 0 ) {
1395 if( guy.get_focus() <= 0 ) {
1396 return 0.0f;
1397 }
1398 float concentration_loss = ( 1.0f - ( guy.get_focus() / 100.0f ) ) *
1399 temp_concentration_difficulty_multiplyer;
1400 fail_chance /= 1.0f - concentration_loss;
1401 psi_fail_chance /= 1.0f - concentration_loss;
1402 }
1403
1404 return clamp( is_psi ? psi_fail_chance : fail_chance, 0.0f, 1.0f );
1405}
