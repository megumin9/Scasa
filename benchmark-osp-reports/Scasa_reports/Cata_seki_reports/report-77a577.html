922bool can_construct( const construction &con, const tripoint_bub_ms &p )
923{
924 const map &here = get_map();
925 const furn_id f = here.furn( p );
926 const ter_id t = here.ter( p );
927 if( con.pre_specials.size() > 1 ) { // pre-functions
928 for( const auto &special : con.pre_specials ) {
929 if( !special( p ) ) {
930 return false;
931 }
932 }
933 } else if( !con.pre_special( p ) ) { // pre-function
934 return false;
935 }
936 if( !has_pre_terrain( con, p ) || // terrain type
937 !can_construct_furn_ter( con, f, t ) ) { // flags
938 return false;
939 }
940 if( !con.post_terrain.empty() ) { // make sure the construction would actually do something
941 if( con.post_is_furniture ) {
942 return f != furn_id( con.post_terrain );
943 } else {
944 return t != ter_id( con.post_terrain );
945 }
946 }
947 return true;
948}
