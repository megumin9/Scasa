226/**
227 * Linear interpolation: returns first argument if t is 0, second if t is 1, otherwise proportional to t.
228 * Does not clamp t, meaning it can return values lower than min (if t<0) or higher than max (if t>1).
229 */
230template<typename T>
231constexpr T lerp( const T &min, const T &max, float t )
232{
233 return ( 1.0f - t ) * min + t * max;
234}

236/** Linear interpolation with t clamped to [0, 1] */
237template<typename T>
238constexpr T lerp_clamped( const T &min, const T &max, float t )
239{
240 return lerp( min, max, clamp( t, 0.0f, 1.0f ) );
241}

243// Inverse of \p lerp, unbounded so it may extrapolate, returns 0.0f if min == max
244// @returns linear factor for interpolating between \p min and \p max to reach \p value
245template<typename T>
246constexpr float inverse_lerp( const T &min, const T &max, const T &value )
247{
248 if( max == min ) {
249 return 0.0f; // avoids a NaN
250 }
251 return ( value - min ) / ( max - min );
252}

254// Remaps \p value from range of \p i_min to \p i_max to a range between \p o_min and \p o_max
255// uses unclamped linear interpolation, so output value may be beyond output range if value is
256// outside input range.
257template<typename Tin, typename Tout>
258constexpr Tout linear_remap( const Tin &i_min, const Tin &i_max,
259 const Tout &o_min, const Tout &o_max, Tin value )
260{
261 const float t = inverse_lerp( i_min, i_max, value );
262 return lerp( o_min, o_max, t );
263}

273/**
274 * @brief Class used to access a list as if it were circular.
275 *
276 * Some times you just want to have a list loop around on itself.
277 * This wrapper class allows you to do that. It requires the list to exist
278 * separately, but that also means any changes to the list get propagated (both ways).
279 */
280template<typename T>
281class list_circularizer
282{
283 private:
284 unsigned int _index = 0;
285 std::vector<T> *_list;
286 public:
287 /** Construct list_circularizer from an existing std::vector. */
288 explicit list_circularizer( std::vector<T> &_list ) : _list( &_list ) {
289 }
290
291 /** Advance list to next item, wrapping back to 0 at end of list */
292 void next() {
293 _index = ( _index == _list->size() - 1 ? 0 : _index + 1 );
294 }
295
296 /** Advance list to previous item, wrapping back to end at zero */
297 void prev() {
298 _index = ( _index == 0 ? _list->size() - 1 : _index - 1 );
299 }
300
301 /** Return list element at the current location */
302 T &cur() const {
303 // list could be null, but it would be a design time mistake and really, the callers fault.
304 return ( *_list )[_index];
305 }
306};

567/**
568 * Checks if two sets are equal, ignoring specified elements.
569 * Works as if `ignored_elements` were temporarily erased from both sets before comparison.
570 * @tparam Set type of the set (must be ordered, i.e. std::set, cata::flat_set)
571 * @param set first set to compare
572 * @param set2 second set to compare
573 * @param ignored_elements elements from both sets to ignore
574 * @return true, if sets without ignored elements are equal, false otherwise
575 */
576template<typename Set, typename T = std::decay_t<decltype( *std::declval<const Set &>().begin() )>>
577bool equal_ignoring_elements( const Set &set, const Set &set2, const Set &ignored_elements )
578{
579 // general idea: splits both sets into the ranges bounded by elements from `ignored_elements`
580 // and checks that these ranges are equal
581
582 // traverses ignored elements in
583 if( ignored_elements.empty() ) {
584 return set == set2;
585 }
586
587 using Iter = typename Set::iterator;
588 Iter end = ignored_elements.end();
589 Iter cur = ignored_elements.begin();
590 Iter prev = cur;
591 cur++;
592
593 // first comparing the sets range [set.begin() .. ignored_elements.begin()]
594 if( !std::equal( set.begin(), set.lower_bound( *prev ),
595 set2.begin(), set2.lower_bound( *prev ) ) ) {
596 return false;
597 }
598
599 // compare ranges bounded by two elements: [`prev` .. `cur`]
600 while( cur != end ) {
601 if( !std::equal( set.upper_bound( *prev ), set.lower_bound( *cur ),
602 set2.upper_bound( *prev ), set2.lower_bound( *cur ) ) ) {
603 return false;
604 }
605 prev = cur;
606 cur++;
607 }
608
609 // compare the range after the last element of ignored_elements: [ignored_elements.back() .. set.end()]
610 return static_cast<bool>( std::equal( set.upper_bound( *prev ), set.end(),
611 set2.upper_bound( *prev ), set2.end() ) );
612}

627template<typename Map, typename Set>
628bool map_equal_ignoring_keys( const Map &lhs, const Map &rhs, const Set &ignore_keys )
629{
630 // Since map and set are sorted, we can do this as a single pass with only conditional checks into remove_keys
631 if( ignore_keys.empty() ) {
632 return lhs == rhs;
633 }
634
635 auto lbegin = lhs.begin();
636 auto lend = lhs.end();
637 auto rbegin = rhs.begin();
638 auto rend = rhs.end();
639
640 for( ; lbegin != lend && rbegin != rend; ++lbegin, ++rbegin ) {
641 // Sanity check keys
642 if( lbegin->first != rbegin->first ) {
643 while( lbegin != lend && ignore_keys.count( lbegin->first ) == 1 ) {
644 ++lbegin;
645 }
646 if( lbegin == lend ) {
647 break;
648 }
649 if( rbegin->first != lbegin->first ) {
650 while( rbegin != rend && ignore_keys.count( rbegin->first ) == 1 ) {
651 ++rbegin;
652 }
653 if( rbegin == rend ) {
654 break;
655 }
656 }
657 // If we've skipped ignored keys and the keys still don't match,
658 // then the maps are unequal.
659 if( lbegin->first != rbegin->first ) {
660 return false;
661 }
662 }
663 if( lbegin->second != rbegin->second && ignore_keys.count( lbegin->first ) != 1 ) {
664 return false;
665 }
666 // Either the values were equal, or the key was ignored.
667 }
668 // At least one map ran out of keys. The other may still have ignored keys in it.
669 while( lbegin != lend && ignore_keys.count( lbegin->first ) ) {
670 ++lbegin;
671 }
672 while( rbegin != rend && ignore_keys.count( rbegin->first ) ) {
673 ++rbegin;
674 }
675 return lbegin == lend && rbegin == rend;
676}
