91// Hoisted to header and inlined so the test in tests/shadowcasting_test.cpp can use it.
92// Beer-Lambert law says attenuation is going to be equal to
93// 1 / (e^al) where a = coefficient of absorption and l = length.
94// Factoring out length, we get 1 / (e^((a1*a2*a3*...*an)*l))
95// We merge all of the absorption values by taking their cumulative average.
96inline float sight_calc( const float &numerator, const float &transparency, const int &distance )
97{
98 return numerator / std::exp( transparency * distance );
99}
100inline bool sight_check( const float &transparency, const float &/*intensity*/ )
101{
102 return transparency > LIGHT_TRANSPARENCY_SOLID;
103}
