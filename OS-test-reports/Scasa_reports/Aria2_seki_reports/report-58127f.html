197size_t MultiDiskAdaptor::tryCloseFile(size_t numClose)
198{
199 size_t left = numClose;
200 for (; !openedDiskWriterEntries_.empty() && left > 0; --left) {
201 // Choose one DiskWriterEntry randomly and close it.
202 size_t index = SimpleRandomizer::getInstance()->getRandomNumber(
203 openedDiskWriterEntries_.size());
204 auto i = std::begin(openedDiskWriterEntries_);
205 std::advance(i, index);
206 (*i)->closeFile();
207 (*i) = openedDiskWriterEntries_.back();
208 openedDiskWriterEntries_.pop_back();
209 }
210 return numClose - left;
211}

337void MultiDiskAdaptor::writeData(const unsigned char* data, size_t len,
338 int64_t offset)
339{
340 auto first = findFirstDiskWriterEntry(diskWriterEntries_, offset);
341 ssize_t rem = len;
342 int64_t fileOffset = offset - (*first)->getFileEntry()->getOffset();
343 for (auto i = first, eoi = diskWriterEntries_.cend(); i != eoi; ++i) {
344 ssize_t writeLength = calculateLength((*i).get(), fileOffset, rem);
345 openIfNot((*i).get(), &DiskWriterEntry::openFile);
346 if (!(*i)->isOpen()) {
347 throwOnDiskWriterNotOpened((*i).get(), offset + (len - rem));
348 }
349
350 (*i)->getDiskWriter()->writeData(data + (len - rem), writeLength,
351 fileOffset);
352 rem -= writeLength;
353 fileOffset = 0;
354 if (rem == 0) {
355 break;
356 }
357 }
