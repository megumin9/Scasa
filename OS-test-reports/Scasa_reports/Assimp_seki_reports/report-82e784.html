542 if (static_cast<size_t>(i) >= tempData.size()) {
543 DOMError("index out of range",&GetRequiredElement(source,indexDataElementName));
544 }
545
546 data_out[next++] = tempData[i];
547 }
548 }
549 else {
550 FBXImporter::LogError("ignoring vertex data channel, access type not implemented: ",
551 MappingInformationType, ",", ReferenceInformationType);
552 }
553}

636// ------------------------------------------------------------------------------------------------
637void MeshGeometry::ReadVertexDataMaterials(std::vector<int>& materials_out, const Scope& source,
638 const std::string& MappingInformationType,
639 const std::string& ReferenceInformationType)
640{
641 const size_t face_count = m_faces.size();
642 if( 0 == face_count )
643 {
644 return;
645 }
646
647 // materials are handled separately. First of all, they are assigned per-face
648 // and not per polyvert. Secondly, ReferenceInformationType=IndexToDirect
649 // has a slightly different meaning for materials.
650 ParseVectorDataArray(materials_out,GetRequiredElement(source,"Materials"));
651
652 if (MappingInformationType == "AllSame") {
653 // easy - same material for all faces
654 if (materials_out.empty()) {
655 FBXImporter::LogError("expected material index, ignoring");
656 return;
657 } else if (materials_out.size() > 1) {
658 FBXImporter::LogWarn("expected only a single material index, ignoring all except the first one");
659 materials_out.clear();
660 }
661
662 materials_out.resize(m_vertices.size());
663 std::fill(materials_out.begin(), materials_out.end(), materials_out.at(0));
664 } else if (MappingInformationType == "ByPolygon" && ReferenceInformationType == "IndexToDirect") {
665 materials_out.resize(face_count);
666
667 if(materials_out.size() != face_count) {
668 FBXImporter::LogError("length of input data unexpected for ByPolygon mapping: ",
669 materials_out.size(), ", expected ", face_count
670 );
671 return;
672 }
673 } else {
674 FBXImporter::LogError("ignoring material assignments, access type not implemented: ",
675 MappingInformationType, ",", ReferenceInformationType);
676 }
677}
