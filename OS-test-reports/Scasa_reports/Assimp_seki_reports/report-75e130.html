458// ------------------------------------------------------------------------------------------------
459void LWOImporter::ComputeNormals(aiMesh *mesh, const std::vector<unsigned int> &smoothingGroups,
460 const LWO::Surface &surface) {
461 // Allocate output storage
462 mesh->mNormals = new aiVector3D[mesh->mNumVertices];
463
464 // First generate per-face normals
465 aiVector3D *out;
466 std::vector<aiVector3D> faceNormals;
467
468 // ... in some cases that's already enough
469 if (!surface.mMaximumSmoothAngle)
470 out = mesh->mNormals;
471 else {
472 faceNormals.resize(mesh->mNumVertices);
473 out = &faceNormals[0];
474 }
475
476 aiFace *begin = mesh->mFaces, *const end = mesh->mFaces + mesh->mNumFaces;
477 for (; begin != end; ++begin) {
478 aiFace &face = *begin;
479
480 if (face.mNumIndices < 3) {
481 continue;
482 }
483
484 // LWO doc: "the normal is defined as the cross product of the first and last edges"
485 aiVector3D *pV1 = mesh->mVertices + face.mIndices[0];
486 aiVector3D *pV2 = mesh->mVertices + face.mIndices[1];
487 aiVector3D *pV3 = mesh->mVertices + face.mIndices[face.mNumIndices - 1];
488
489 aiVector3D vNor = ((*pV2 - *pV1) ^ (*pV3 - *pV1)).Normalize();
490 for (unsigned int i = 0; i < face.mNumIndices; ++i)
491 out[face.mIndices[i]] = vNor;
492 }
493 if (!surface.mMaximumSmoothAngle) return;
494 const float posEpsilon = ComputePositionEpsilon(mesh);
495
496 // Now generate the spatial sort tree
497 SGSpatialSort sSort;
498 std::vector<unsigned int>::const_iterator it = smoothingGroups.begin();
499 for (begin = mesh->mFaces; begin != end; ++begin, ++it) {
500 aiFace &face = *begin;
501 for (unsigned int i = 0; i < face.mNumIndices; ++i) {
502 unsigned int tt = face.mIndices[i];
503 sSort.Add(mesh->mVertices[tt], tt, *it);
504 }
505 }
506 // Sort everything - this takes O(nlogn) time
507 sSort.Prepare();
508 std::vector<unsigned int> poResult;
509 poResult.reserve(20);
510
511 // Generate vertex normals. We have O(logn) for the binary lookup, which we need
512 // for n elements, thus the EXPECTED complexity is O(nlogn)
513 if (surface.mMaximumSmoothAngle < 3.f && !configSpeedFlag) {
514 const float fLimit = std::cos(surface.mMaximumSmoothAngle);
515
516 for (begin = mesh->mFaces, it = smoothingGroups.begin(); begin != end; ++begin, ++it) {
517 const aiFace &face = *begin;
518 unsigned int *beginIdx = face.mIndices, *const endIdx = face.mIndices + face.mNumIndices;
519 for (; beginIdx != endIdx; ++beginIdx) {
520 unsigned int idx = *beginIdx;
521 sSort.FindPositions(mesh->mVertices[idx], *it, posEpsilon, poResult, true);
522
523 aiVector3D vNormals;
524 for (std::vector<unsigned int>::const_iterator a = poResult.begin(); a != poResult.end(); ++a) {
525 const aiVector3D &v = faceNormals[*a];
526 if (v * faceNormals[idx] < fLimit)
527 continue;
528 vNormals += v;
529 }
530 }
531 }
532 }
533 // faster code path in case there is no smooth angle
534 else {
535 std::vector<bool> vertexDone(mesh->mNumVertices, false);
536 for (begin = mesh->mFaces, it = smoothingGroups.begin(); begin != end; ++begin, ++it) {
537 const aiFace &face = *begin;
538 unsigned int *beginIdx = face.mIndices, *const endIdx = face.mIndices + face.mNumIndices;
539 for (; beginIdx != endIdx; ++beginIdx) {
540 unsigned int idx = *beginIdx;
541 if (vertexDone[idx])
542 continue;
543 sSort.FindPositions(mesh->mVertices[idx], *it, posEpsilon, poResult, true);
544
545 aiVector3D vNormals;
546 for (std::vector<unsigned int>::const_iterator a = poResult.begin(); a != poResult.end(); ++a) {
547 const aiVector3D &v = faceNormals[*a];
548 vNormals += v;
549 }
550 for (std::vector<unsigned int>::const_iterator a = poResult.begin(); a != poResult.end(); ++a) {
551 mesh->mNormals[*a] = vNormals;
552 vertexDone[*a] = true;
553 }
554 }
555 }
556 }
557 GeometryUtils::normalizeVectorArray(mesh->mNormals, mesh->mNormals, mesh->mNumVertices);
558}

1367void LWOImporter::LoadLWO3Envelope(unsigned int length) {
1368 LE_NCONST uint8_t *const end = mFileBuffer + length;
1369 AI_LWO_VALIDATE_CHUNK_LENGTH(length, ENVL, 4)if (length < 4) { throw DeadlyImportError("LWO: " "ENVL" " chunk is too small"
); };
1370
1371 mEnvelopes.emplace_back();
1372 LWO::Envelope &envelope = mEnvelopes.back();
1373
1374 // Get the index of the envelope
1375 envelope.index = ReadVSizedIntLWO2(mFileBuffer);
1376
1377 // ... and read all blocks
1378 while (true) {
1379 if (mFileBuffer + 8 >= end) break;
1380 LE_NCONST IFF::ChunkHeader head = IFF::LoadChunk(mFileBuffer);
1381
1382 if (mFileBuffer + head.length > end)
1383 throw DeadlyImportError("LWO3: Invalid envelope chunk length");
1384
1385 uint8_t *const next = mFileBuffer + head.length;
1386 switch (head.type) {
1387 // Type & representation of the envelope
1388 case AI_LWO_TYPE((uint32_t) (((uint8_t)'T' << 24u) | ((uint8_t)'Y' <<
16u) | ((uint8_t)'P' << 8u) | ((uint8_t)'E'))):
1389 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, TYPE, 4)if (head.length < 4) { throw DeadlyImportError("LWO: " "TYPE"
" chunk is too small"); };
1390 mFileBuffer++; // skip user format
1391
1392 // Determine type of envelope
1393 envelope.type = (LWO::EnvelopeType)*mFileBuffer;
1394 ++mFileBuffer;
1395 break;
1396
1397 // precondition
1398 case AI_LWO_PRE((uint32_t) (((uint8_t)'P' << 24u) | ((uint8_t)'R' <<
16u) | ((uint8_t)'E' << 8u) | ((uint8_t)' '))):
1399 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, PRE, 4)if (head.length < 4) { throw DeadlyImportError("LWO: " "PRE"
" chunk is too small"); };
1400 envelope.pre = (LWO::PrePostBehaviour)GetU2();
1401 break;
1402
1403 // postcondition
1404 case AI_LWO_POST((uint32_t) (((uint8_t)'P' << 24u) | ((uint8_t)'O' <<
16u) | ((uint8_t)'S' << 8u) | ((uint8_t)'T'))):
1405 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, POST, 4)if (head.length < 4) { throw DeadlyImportError("LWO: " "POST"
" chunk is too small"); };
1406 envelope.post = (LWO::PrePostBehaviour)GetU2();
1407 break;
1408
1409 // keyframe
1410 case AI_LWO_KEY((uint32_t) (((uint8_t)'K' << 24u) | ((uint8_t)'E' <<
16u) | ((uint8_t)'Y' << 8u) | ((uint8_t)' '))): {
1411 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, KEY, 10)if (head.length < 10) { throw DeadlyImportError("LWO: " "KEY"
" chunk is too small"); };
1412
1413 envelope.keys.emplace_back();
1414 LWO::Key &key = envelope.keys.back();
1415
1416 key.time = GetF4();
1417 key.value = GetF4();
1418 break;
1419 }
1420
1421 // interval interpolation
1422 case AI_LWO_SPAN((uint32_t) (((uint8_t)'S' << 24u) | ((uint8_t)'P' <<
16u) | ((uint8_t)'A' << 8u) | ((uint8_t)'N'))): {
1423 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, SPAN, 6)if (head.length < 6) { throw DeadlyImportError("LWO: " "SPAN"
" chunk is too small"); };
1424 if (envelope.keys.size() < 2)
1425 ASSIMP_LOG_WARN("LWO3: Unexpected SPAN chunk")Assimp::DefaultLogger::get()->warn("LWO3: Unexpected SPAN chunk"
);
1426 else {
1427 LWO::Key &key = envelope.keys.back();
1428 switch (GetU4()) {
1429 case AI_LWO_STEP((uint32_t) (((uint8_t)'S' << 24u) | ((uint8_t)'T' <<
16u) | ((uint8_t)'E' << 8u) | ((uint8_t)'P'))):
1430 key.inter = LWO::IT_STEP;
1431 break;
1432 case AI_LWO_LINE((uint32_t) (((uint8_t)'L' << 24u) | ((uint8_t)'I' <<
16u) | ((uint8_t)'N' << 8u) | ((uint8_t)'E'))):
1433 key.inter = LWO::IT_LINE;
1434 break;
1435 case AI_LWO_TCB((uint32_t) (((uint8_t)'T' << 24u) | ((uint8_t)'C' <<
16u) | ((uint8_t)'B' << 8u) | ((uint8_t)' '))):
1436 key.inter = LWO::IT_TCB;
1437 break;
1438 case AI_LWO_HERM((uint32_t) (((uint8_t)'H' << 24u) | ((uint8_t)'E' <<
16u) | ((uint8_t)'R' << 8u) | ((uint8_t)'M'))):
1439 key.inter = LWO::IT_HERM;
1440 break;
1441 case AI_LWO_BEZI((uint32_t) (((uint8_t)'B' << 24u) | ((uint8_t)'E' <<
16u) | ((uint8_t)'Z' << 8u) | ((uint8_t)'I'))):
1442 key.inter = LWO::IT_BEZI;
1443 break;
1444 case AI_LWO_BEZ2((uint32_t) (((uint8_t)'B' << 24u) | ((uint8_t)'E' <<
16u) | ((uint8_t)'Z' << 8u) | ((uint8_t)'2'))):
1445 key.inter = LWO::IT_BEZ2;
1446 break;
1447 default:
1448 ASSIMP_LOG_WARN("LWO3: Unknown interval interpolation mode")Assimp::DefaultLogger::get()->warn("LWO3: Unknown interval interpolation mode"
);
1449 };
1450
1451 // todo ... read params
1452 }
1453 break;
1454 }
1455
1456 default:
1457 ASSIMP_LOG_WARN("LWO3: Encountered unknown ENVL subchunk")Assimp::DefaultLogger::get()->warn("LWO3: Encountered unknown ENVL subchunk"
);
1458 break;
1459 }
1460 // regardless how much we did actually read, go to the next chunk
1461 mFileBuffer = next;
1462 }

1467void LWOImporter::LoadLWO2File() {
1468 bool skip = false;
1469
1470 LE_NCONST uint8_t *const end = mFileBuffer + fileSize;
1471 unsigned int iUnnamed = 0;
1472
1473 while (true) {
1474 if (mFileBuffer + sizeof(IFF::ChunkHeader) > end) break;
1475
1476 IFF::ChunkHeader head = IFF::LoadChunk(mFileBuffer);
1477
1478 int bufOffset = 0;
1479 if( head.type == AI_IFF_FOURCC_FORM((uint32_t) (((uint8_t)'F' << 24u) | ((uint8_t)'O' <<
16u) | ((uint8_t)'R' << 8u) | ((uint8_t)'M'))) ) { // not chunk, it's a form
1480 mFileBuffer -= 8;
1481 head = IFF::LoadForm(mFileBuffer);
1482 bufOffset = 4;
1483 }
1484
1485 if (mFileBuffer + head.length > end) {
1486 throw DeadlyImportError("LWO2: Chunk length points behind the file");
1487 }
1488 uint8_t *const next = mFileBuffer + head.length;
1489 mFileBuffer += bufOffset;
1490 if (!head.length) {
1491 mFileBuffer = next;
1492 continue;
1493 }
1494
1495 switch (head.type) {
1496 // new layer
1497 case AI_LWO_LAYR((uint32_t) (((uint8_t)'L' << 24u) | ((uint8_t)'A' <<
16u) | ((uint8_t)'Y' << 8u) | ((uint8_t)'R'))): {
1498 // add a new layer to the list ....
1499 mLayers->push_back(LWO::Layer());
1500 LWO::Layer &layer = mLayers->back();
1501 mCurLayer = &layer;
1502
1503 AI_LWO_VALIDATE_CHUNK_LENGTH(head.length, LAYR, 16)if (head.length < 16) { throw DeadlyImportError("LWO: " "LAYR"
" chunk is too small"); };
1504
1505 // layer index.
1506 layer.mIndex = GetU2();
1507
1508 // Continue loading this layer or ignore it? Check the layer index property
1509 if (UINT_MAX(2147483647 *2U +1U) != configLayerIndex && (configLayerIndex - 1) != layer.mIndex) {
1510 skip = true;
1511 } else
1512 skip = false;
1513
1514 // pivot point
1515 mFileBuffer += 2; /* unknown */
1516 mCurLayer->mPivot.x = GetF4();
1517 mCurLayer->mPivot.y = GetF4();
1518 mCurLayer->mPivot.z = GetF4();
1519 GetS0(layer.mName, head.length - 16);
1520
1521 // if the name is empty, generate a default name
1522 if (layer.mName.empty()) {
1523 char buffer[128]; // should be sufficiently large
1524 ::ai_snprintfsnprintf(buffer, 128, "Layer_%i", iUnnamed++);
1525 layer.mName = buffer;
1526 }
1527
1528 // load this layer or ignore it? Check the layer name property
1529 if (configLayerName.length() && configLayerName != layer.mName) {
1530 skip = true;
1531 } else
1532 hasNamedLayer = true;
1533
1534 // optional: parent of this layer
1535 if (mFileBuffer + 2 <= next)
1536 layer.mParent = GetU2();
1537 else
1538 layer.mParent = (uint16_t) -1;
1539
1540 // Set layer skip parameter
1541 layer.skip = skip;
1542
1543 break;
1544 }
1545 // vertex list
1546 case AI_LWO_PNTS((uint32_t) (((uint8_t)'P' << 24u) | ((uint8_t)'N' <<
16u) | ((uint8_t)'T' << 8u) | ((uint8_t)'S'))): {
1547 if (skip)
1548 break;
1549
1550 unsigned int old = (unsigned int)mCurLayer->mTempPoints.size();
1551 LoadLWOPoints(head.length);
1552 mCurLayer->mPointIDXOfs = old;
1553 break;
1554 }
1555 // vertex tags
1556 case AI_LWO_VMAD((uint32_t) (((uint8_t)'V' << 24u) | ((uint8_t)'M' <<
16u) | ((uint8_t)'A' << 8u) | ((uint8_t)'D'))):
1557 if (mCurLayer->mFaces.empty()) {
1558 ASSIMP_LOG_WARN("LWO2: Unexpected VMAD chunk")Assimp::DefaultLogger::get()->warn("LWO2: Unexpected VMAD chunk"
);
1559 break;
1560 }
1561 // --- intentionally no break here
1562 // fallthrough
1563 case AI_LWO_VMAP((uint32_t) (((uint8_t)'V' << 24u) | ((uint8_t)'M' <<
16u) | ((uint8_t)'A' << 8u) | ((uint8_t)'P'))): {
1564 if (skip)
1565 break;
1566
1567 if (mCurLayer->mTempPoints.empty())
1568 ASSIMP_LOG_WARN("LWO2: Unexpected VMAP chunk")Assimp::DefaultLogger::get()->warn("LWO2: Unexpected VMAP chunk"
);
1569 else
1570 LoadLWO2VertexMap(head.length, head.type == AI_LWO_VMAD((uint32_t) (((uint8_t)'V' << 24u) | ((uint8_t)'M' <<
16u) | ((uint8_t)'A' << 8u) | ((uint8_t)'D'))));
1571 break;
1572 }
1573 // face list
1574 case AI_LWO_POLS((uint32_t) (((uint8_t)'P' << 24u) | ((uint8_t)'O' <<
16u) | ((uint8_t)'L' << 8u) | ((uint8_t)'S'))): {
1575 if (skip)
1576 break;
1577
1578 unsigned int old = (unsigned int)mCurLayer->mFaces.size();
1579 LoadLWO2Polygons(head.length);
1580 mCurLayer->mFaceIDXOfs = old;
1581 break;
1582 }
1583 // polygon tags
1584 case AI_LWO_PTAG((uint32_t) (((uint8_t)'P' << 24u) | ((uint8_t)'T' <<
16u) | ((uint8_t)'A' << 8u) | ((uint8_t)'G'))): {
1585 if (skip)

1587
1588 if (mCurLayer->mFaces.empty()) {
1589 ASSIMP_LOG_WARN("LWO2: Unexpected PTAG")Assimp::DefaultLogger::get()->warn("LWO2: Unexpected PTAG"
);
1590 } else {
1591 LoadLWO2PolygonTags(head.length);
1592 }
1593 break;
1594 }
1595 // list of tags
1596 case AI_LWO_TAGS((uint32_t) (((uint8_t)'T' << 24u) | ((uint8_t)'A' <<
16u) | ((uint8_t)'G' << 8u) | ((uint8_t)'S'))): {
1597 if (!mTags->empty()) {
1598 ASSIMP_LOG_WARN("LWO2: SRFS chunk encountered twice")Assimp::DefaultLogger::get()->warn("LWO2: SRFS chunk encountered twice"
);
1599 } else {
1600 LoadLWOTags(head.length);
1601 }
1602 break;
1603 }
1604
1605 // surface chunk
1606 case AI_LWO_SURF((uint32_t) (((uint8_t)'S' << 24u) | ((uint8_t)'U' <<
16u) | ((uint8_t)'R' << 8u) | ((uint8_t)'F'))): {
1607 if( mIsLWO3 )
1608 LoadLWO3Surface(head.length);
1609 else
1610 LoadLWO2Surface(head.length);
1611
1612 break;
1613 }
1614
1615 // clip chunk
1616 case AI_LWO_CLIP((uint32_t) (((uint8_t)'C' << 24u) | ((uint8_t)'L' <<
16u) | ((uint8_t)'I' << 8u) | ((uint8_t)'P'))): {
1617 if( mIsLWO3 )
1618 LoadLWO3Clip(head.length);
1619 else
1620 LoadLWO2Clip(head.length);
1621 break;
1622 }

1624 // envelope chunk
1625 case AI_LWO_ENVL((uint32_t) (((uint8_t)'E' << 24u) | ((uint8_t)'N' <<
16u) | ((uint8_t)'V' << 8u) | ((uint8_t)'L'))): {
1626 if( mIsLWO3 )
1627 LoadLWO3Envelope(head.length);
1628 else
1629 LoadLWO2Envelope(head.length);
1630 break;
1631 }
1632 }
1633 mFileBuffer = next;
1634 }
