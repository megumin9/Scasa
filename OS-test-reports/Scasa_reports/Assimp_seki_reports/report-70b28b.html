562// Check if m0,m1 intersects n0,n1 assuming same ordering of the points in the line segments
563// output the intersection points on n0,n1
564bool IntersectingLineSegments(const IfcVector2& n0, const IfcVector2& n1,
565 const IfcVector2& m0, const IfcVector2& m1,
566 IfcVector2& out0, IfcVector2& out1) {
567 const IfcVector2 n0_to_n1 = n1 - n0;
568
569 const IfcVector2 n0_to_m0 = m0 - n0;
570 const IfcVector2 n1_to_m1 = m1 - n1;
571
572 const IfcVector2 n0_to_m1 = m1 - n0;
573
574 const IfcFloat e = 1e-5f;
575 const IfcFloat smalle = 1e-9f;
576
577 static const IfcFloat inf = std::numeric_limits<IfcFloat>::infinity();
578
579 if (!(n0_to_m0.SquareLength() < e*e || std::fabs(n0_to_m0 * n0_to_n1) / (n0_to_m0.Length() * n0_to_n1.Length()) > 1-1e-5 )) {
580 return false;
581 }
582
583 if (!(n1_to_m1.SquareLength() < e*e || std::fabs(n1_to_m1 * n0_to_n1) / (n1_to_m1.Length() * n0_to_n1.Length()) > 1-1e-5 )) {
584 return false;
585 }
586
587 IfcFloat s0;
588 IfcFloat s1;
589
590 // pick the axis with the higher absolute difference so the result
591 // is more accurate. Since we cannot guarantee that the axis with
592 // the higher absolute difference is big enough as to avoid
593 // divisions by zero, the case 0/0 ~ infinity is detected and
594 // handled separately.
595 if(std::fabs(n0_to_n1.x) > std::fabs(n0_to_n1.y)) {
596 s0 = n0_to_m0.x / n0_to_n1.x;
597 s1 = n0_to_m1.x / n0_to_n1.x;
598
599 if (std::fabs(s0) == inf && std::fabs(n0_to_m0.x) < smalle) {
600 s0 = 0.;
601 }
602 if (std::fabs(s1) == inf && std::fabs(n0_to_m1.x) < smalle) {
603 s1 = 0.;
604 }
605 } else {
606 s0 = n0_to_m0.y / n0_to_n1.y;
607 s1 = n0_to_m1.y / n0_to_n1.y;
608
609 if (std::fabs(s0) == inf && std::fabs(n0_to_m0.y) < smalle) {
610 s0 = 0.;
611 }
612 if (std::fabs(s1) == inf && std::fabs(n0_to_m1.y) < smalle) {
613 s1 = 0.;
614 }
615 }
616
617 if (s1 < s0) {
618 std::swap(s1,s0);
619 }
620
621 s0 = std::max(0.0,s0);
622 s1 = std::max(0.0,s1);
623
624 s0 = std::min(1.0,s0);
625 s1 = std::min(1.0,s1);
626
627 if (std::fabs(s1-s0) < e) {
628 return false;
629 }
630
631 out0 = n0 + s0 * n0_to_n1;
632 out1 = n0 + s1 * n0_to_n1;
633
634 return true;
635}

1083 // rescale them to [0,1] range. This normalization means all further
1084 // epsilons need not be scaled.
1085 bool ok = true;
1086
1087 std::vector<IfcVector2> contour_flat;
1088
1089 IfcVector3 nor;
1090 const IfcMatrix4 m = ProjectOntoPlane(contour_flat, curmesh, ok, nor);
1091 if(!ok) {
1092 return false;
1093 }
1094
1095 // Obtain inverse transform for getting back to world space later on
1096 const IfcMatrix4 minv = IfcMatrix4(m).Inverse();
1097
1098 // Compute bounding boxes for all 2D openings in projection space
1099 ContourVector contours;
1100
1101 std::vector<IfcVector2> temp_contour;
1102 std::vector<IfcVector2> temp_contour2;
1103
1104 IfcVector3 wall_extrusion_axis_norm = wall_extrusion_axis;
1105 wall_extrusion_axis_norm.Normalize();
1106
1107 for(TempOpening& opening :openings) {
1108
1109 // extrusionDir may be 0,0,0 on case where the opening mesh is not an
1110 // IfcExtrudedAreaSolid but something else (i.e. a brep)
1111 IfcVector3 norm_extrusion_dir = opening.extrusionDir;
1112 if (norm_extrusion_dir.SquareLength() > 1e-10) {
1113 norm_extrusion_dir.Normalize();
1114 } else {
1115 norm_extrusion_dir = IfcVector3();
1116 }
1117
1118 TempMesh* profile_data = opening.profileMesh.get();
1119 bool is_2d_source = false;
1120 if (opening.profileMesh2D && norm_extrusion_dir.SquareLength() > 0) {
1121 if (std::fabs(norm_extrusion_dir * nor) > 0.9) {
1122 profile_data = opening.profileMesh2D.get();
1123 is_2d_source = true;
1124 }
1125 }
1126 std::vector<IfcVector3> profile_verts = profile_data->mVerts;
1127 std::vector<unsigned int> profile_vertcnts = profile_data->mVertcnt;
1128 if(profile_verts.size() <= 2) {
1129 continue;
1130 }
1131
1132 // The opening meshes are real 3D meshes so skip over all faces
1133 // clearly facing into the wrong direction. Also, we need to check
1134 // whether the meshes do actually intersect the base surface plane.
1135 // This is done by recording minimum and maximum values for the
1136 // d component of the plane equation for all polys and checking
1137 // against surface d.
1138
1139 // Use the sign of the dot product of the face normal to the plane
1140 // normal to determine to which side of the difference mesh a
1141 // triangle belongs. Get independent bounding boxes and vertex
1142 // sets for both sides and take the better one (we can't just
1143 // take both - this would likely cause major screwup of vertex
1144 // winding, producing errors as late as in CloseWindows()).
1145 IfcFloat dmin, dmax;
1146 MinMaxChooser<IfcFloat>()(dmin,dmax);
1147
1148 temp_contour.clear();
1149 temp_contour2.clear();
1150
1151 IfcVector2 vpmin,vpmax;
1152 MinMaxChooser<IfcVector2>()(vpmin,vpmax);
1153
1154 IfcVector2 vpmin2,vpmax2;
1155 MinMaxChooser<IfcVector2>()(vpmin2,vpmax2);
1156
1157 for (size_t f = 0, vi_total = 0, fend = profile_vertcnts.size(); f < fend; ++f) {
1158
1159 bool side_flag = true;
1160 if (!is_2d_source) {
1161 const IfcVector3 face_nor = ((profile_verts[vi_total+2] - profile_verts[vi_total]) ^
1162 (profile_verts[vi_total+1] - profile_verts[vi_total])).Normalize();
1163
1164 const IfcFloat abs_dot_face_nor = std::abs(nor * face_nor);
1165 if (abs_dot_face_nor < 0.9) {
1166 vi_total += profile_vertcnts[f];
1167 continue;
1168 }
1169
1170 side_flag = nor * face_nor > 0;
1171 }
1172
1173 for (unsigned int vi = 0, vend = profile_vertcnts[f]; vi < vend; ++vi, ++vi_total) {
1174 const IfcVector3& x = profile_verts[vi_total];
1175
1176 const IfcVector3 v = m * x;
1177 IfcVector2 vv(v.x, v.y);
1178
1179 dmin = std::min(dmin, v.z);
1180 dmax = std::max(dmax, v.z);
1181
1182 // sanity rounding
1183 vv = std::max(vv,IfcVector2());
1184 vv = std::min(vv,one_vec);
1185
1186 if(side_flag) {
1187 vpmin = std::min(vpmin,vv);
1188 vpmax = std::max(vpmax,vv);
1189 } else {
1190 vpmin2 = std::min(vpmin2,vv);
1191 vpmax2 = std::max(vpmax2,vv);
1192 }
1193
1194 std::vector<IfcVector2>& store = side_flag ? temp_contour : temp_contour2;
1195
1196 if (!IsDuplicateVertex(vv, store)) {
1197 store.push_back(vv);
1198 }
1199 }
1200 }
1201
1202 if (temp_contour2.size() > 2) {
1203 ai_assert(!is_2d_source)(void)((!!(!is_2d_source)) || (Assimp::aiAssertViolation("!is_2d_source"
, "/usr/STLcheck/sekitest/assimp/code/AssetLib/IFC/IFCOpenings.cpp"
, 1203), 0));
1204 const IfcVector2 area = vpmax-vpmin;
1205 const IfcVector2 area2 = vpmax2-vpmin2;
1206 if (temp_contour.size() <= 2 || std::fabs(area2.x * area2.y) > std::fabs(area.x * area.y)) {
1207 temp_contour.swap(temp_contour2);
1208
1209 vpmax = vpmax2;
1210 vpmin = vpmin2;
1211 }
1212 }
1213 if(temp_contour.size() <= 2) {
1214 continue;
1215 }
1216
1217 // TODO: This epsilon may be too large
1218 const IfcFloat epsilon = std::fabs(dmax-dmin) * 0.0001;
1219 if (!is_2d_source && check_intersection && (0 < dmin-epsilon || 0 > dmax+epsilon)) {
1220 continue;
1221 }
1222
1223 BoundingBox bb = BoundingBox(vpmin,vpmax);
1224
1225 // Skip over very small openings - these are likely projection errors
1226 // (i.e. they don't belong to this side of the wall)
1227 if(std::fabs(vpmax.x - vpmin.x) * std::fabs(vpmax.y - vpmin.y) < static_cast<IfcFloat>(1e-10)) {
1228 continue;
1229 }
1230 std::vector<TempOpening*> joined_openings(1, &opening);
1231
1232 bool is_rectangle = temp_contour.size() == 4;
1233
1234 // See if this BB intersects or is in close adjacency to any other BB we have so far.
1235 for (ContourVector::iterator it = contours.begin(); it != contours.end(); ) {
1236 const BoundingBox& ibb = (*it).bb;
1237 if (BoundingBoxesOverlapping(ibb, bb)) {
1238 if (!(*it).is_rectangular) {
1239 is_rectangle = false;
1240 }
1241
1242 const std::vector<IfcVector2>& other = (*it).contour;
1243 ClipperLib::Paths poly;
1244
1245 // First check whether subtracting the old contour (to which ibb belongs)
1246 // from the new contour (to which bb belongs) yields an updated bb which
1247 // no longer overlaps ibb
1248 MakeDisjunctWindowContours(other, temp_contour, poly);
1249 if(poly.size() == 1) {
1250 const BoundingBox newbb = GetBoundingBox(poly[0]);
1251 if (!BoundingBoxesOverlapping(ibb, newbb )) {
1252 // Good guy bounding box
1253 bb = newbb ;
1254
1255 ExtractVerticesFromClipper(poly[0], temp_contour, false);
1256 continue;
1257 }
1258 }
1259
1260 // Take these two overlapping contours and try to merge them. If they
1261 // overlap (which should not happen, but in fact happens-in-the-real-
1262 // world [tm] ), resume using a single contour and a single bounding box.
1263 MergeWindowContours(temp_contour, other, poly);
1264
1265 if (poly.size() > 1) {
1266 return TryAddOpenings_Poly2Tri(openings, curmesh);
1267 } else if (poly.empty()) {
1268 IFCImporter::LogWarn("ignoring duplicate opening");
1269 temp_contour.clear();
1270 break;
1271 } else {
1272 IFCImporter::LogVerboseDebug("merging overlapping openings");
1273 ExtractVerticesFromClipper(poly[0], temp_contour, false);
1274
1275 // Generate the union of the bounding boxes
1276 bb.first = std::min(bb.first, ibb.first);
1277 bb.second = std::max(bb.second, ibb.second);
1278
1279 // Update contour-to-opening tables accordingly
1280 if (generate_connection_geometry) {
1281 std::vector<TempOpening*>& t = contours_to_openings[std::distance(contours.begin(),it)];
1282 joined_openings.insert(joined_openings.end(), t.begin(), t.end());
1283
1284 contours_to_openings.erase(contours_to_openings.begin() + std::distance(contours.begin(),it));
1285 }
1286
1287 contours.erase(it);
1288
1289 // Restart from scratch because the newly formed BB might now
1290 // overlap any other BB which its constituent BBs didn't
1291 // previously overlap.
1292 it = contours.begin();
1293 continue;
1294 }
1295 }
1296 ++it;
1297 }
1298
1299 if(!temp_contour.empty()) {
1300 if (generate_connection_geometry) {
1301 contours_to_openings.emplace_back(
1302 joined_openings.begin(),
1303 joined_openings.end());
1304 }
1305
1306 contours.emplace_back(temp_contour, bb, is_rectangle);
1307 }
1308 }
1309
1310 // Check if we still have any openings left - it may well be that this is
1311 // not the cause, for example if all the opening candidates don't intersect
1312 // this surface or point into a direction perpendicular to it.
1313 if (contours.empty()) {
1314 return false;
1315 }
1316
1317 curmesh.Clear();
1318
1319 // Generate a base subdivision into quads to accommodate the given list
1320 // of window bounding boxes.
1321 Quadrify(contours,curmesh);
1322
1323 // Run a sanity cleanup pass on the window contours to avoid generating
1324 // artifacts during the contour generation phase later on.
1325 CleanupWindowContours(contours);
1326
1327 // Previously we reduced all windows to rectangular AABBs in projection
1328 // space, now it is time to fill the gaps between the BBs and the real
1329 // window openings.
1330 InsertWindowContours(contours,openings, curmesh);
1331
1332 // Clip the entire outer contour of our current result against the real
1333 // outer contour of the surface. This is necessary because the result
1334 // of the Quadrify() algorithm is always a square area spanning
1335 // over [0,1]^2 (i.e. entire projection space).
1336 CleanupOuterContour(contour_flat, curmesh);
1337
1338 // Undo the projection and get back to world (or local object) space
1339 for(IfcVector3& v3 : curmesh.mVerts) {
1340 v3 = minv * v3;
1341 }
1342
1343 // Generate window caps to connect the symmetric openings on both sides
1344 // of the wall.
1345 if (generate_connection_geometry) {
1346 CloseWindows(contours, minv, contours_to_openings, curmesh);
1347 }
1348 return true;
1349}

1419std::vector<std::vector<IfcVector2>> GetContoursInPlane3D(const std::shared_ptr<TempMesh>& mesh,IfcMatrix3 planeSpace,
1420 IfcFloat planeOffset) {
1421 {
1422 std::stringstream msg;
1423 msg << "GetContoursInPlane3D: planeSpace is \n";
1424 msg << planeSpace.a1 << " " << planeSpace.a2 << " " << planeSpace.a3 << " " << "\n";
1425 msg << planeSpace.b1 << " " << planeSpace.b2 << " " << planeSpace.b3 << " " << "\n";
1426 msg << planeSpace.c1 << " " << planeSpace.c2 << " " << planeSpace.c3 << " " << "\n";
1427 msg << "\n planeOffset is " << planeOffset;
1428 IFCImporter::LogInfo(msg.str().c_str());
1429 }
1430
1431 // we'll put our line segments in here, and then merge them together into contours later
1432 std::deque<std::pair<IfcVector2,IfcVector2>> lineSegments;
1433
1434 // find the lines giving the intersection of the faces with the plane - we'll work in planeSpace throughout.
1435 size_t vI0{ 0 }; // vertex index for first vertex in plane
1436 for(auto nVertices : mesh->mVertcnt) { // iterate over faces
1437 {
1438 std::stringstream msg;
1439 msg << "GetContoursInPlane3D: face (transformed) is \n";
1440 for(auto vI = vI0; vI < vI0 + nVertices; vI++) {
1441 auto v = planeSpace * mesh->mVerts[vI];
1442 msg << " " << v.x << " " << v.y << " " << v.z << " " << "\n";
1443 }
1444 IFCImporter::LogInfo(msg.str().c_str());
1445 }
1446
1447 // not a plane, a point or line
1448 if(nVertices <= 2) {
1449 std::stringstream msg;
1450 msg << "GetContoursInPlane3D: found point or line when expecting plane (only " << nVertices << " vertices)";
1451 IFCImporter::LogWarn(msg.str().c_str());
1452 vI0 += nVertices;
1453 continue;
1454 }
1455
1456 auto v0 = planeSpace * mesh->mVerts[vI0];
1457
1458 // now calculate intersections between face and plane
1459 IfcVector2 firstPoint;
1460 bool gotFirstPoint(false);
1461
1462 if(std::fabs(v0.z - planeOffset) < close) {
1463 // first point is on the plane
1464 firstPoint.x = v0.x;
1465 firstPoint.y = v0.y;
1466 gotFirstPoint = true;
1467 }
1468
1469 auto vn = v0;
1470 for(auto vI = vI0 + 1; vI < vI0 + nVertices; vI++) {
1471 auto vp = vn;
1472 vn = planeSpace * mesh->mVerts[vI];
1473 IfcVector3 intersection;
1474
1475 if(std::fabs(vn.z - planeOffset) < close) {
1476 // on the plane
1477 intersection = vn;
1478 } else if((vn.z > planeOffset) != (vp.z > planeOffset)) {
1479 // passes through the plane
1480 auto vdir = vn - vp;
1481 auto scale = (planeOffset - vp.z) / vdir.z;
1482 intersection = vp + scale * vdir;
1483 } else {
1484 // nowhere near - move on
1485 continue;
1486 }
1487
1488 if(!gotFirstPoint) {
1489 if(std::fabs(vp.z - planeOffset) < close) {
1490 // just had a second line along the plane
1491 firstPoint.x = vp.x;
1492 firstPoint.y = vp.y;
1493 IfcVector2 secondPoint(intersection.x,intersection.y);
1494 auto s = std::pair<IfcVector2,IfcVector2>(firstPoint,secondPoint);
1495 logSegment(s);
1496 lineSegments.push_back(s);
1497 // next firstpoint should be this one
1498 } else {
1499 // store the first intersection point
1500 firstPoint.x = intersection.x;
1501 firstPoint.y = intersection.y;
1502 gotFirstPoint = true;
1503 }
1504 } else {
1505 // now got the second point, so store the pair
1506 IfcVector2 secondPoint(intersection.x,intersection.y);
1507 auto s = std::pair<IfcVector2,IfcVector2>(firstPoint,secondPoint);
1508 logSegment(s);
1509 lineSegments.push_back(s);
1510
1511 // - note that we don't move onto the next face as a non-convex face can create two or more intersections with a plane
1512 gotFirstPoint = false;
1513 }
1514 }
1515 if(gotFirstPoint) {
1516 IFCImporter::LogWarn("GetContoursInPlane3D: odd number of intersections with plane");
1517 }
1518 vI0 += nVertices;
1519 }
1520
1521 {
1522 std::stringstream msg;
1523 msg << "GetContoursInPlane3D: found " << lineSegments.size() << " line segments:\n";
1524 IFCImporter::LogInfo(msg.str().c_str());
1525
1526 for(auto& s : lineSegments) {
1527 logSegment(s);
1528 }
1529
1530 }
1531
1532 // now merge contours until we have the best-looking polygons we can
1533 std::vector<Contour> contours;
1534 while(!lineSegments.empty()) {
1535 // start with a polygon and make the best closed contour we can
1536 const auto& firstSeg = lineSegments.front();
1537 std::deque<IfcVector2> contour{ firstSeg.first, firstSeg.second };
1538 lineSegments.pop_front();
1539 bool foundNextPoint{ true };
1540 bool closedContour{ false };
1541 while(foundNextPoint) {
1542 foundNextPoint = false;
1543 for(auto nextSeg = lineSegments.begin(); nextSeg != lineSegments.end(); nextSeg++) {
1544 // see if we can match up both ends - in which case we've closed the contour
1545 if((isClose(contour.front(),nextSeg->first) && isClose(contour.back(),nextSeg->second)) ||
1546 (isClose(contour.back(),nextSeg->first) && isClose(contour.front(),nextSeg->second))
1547 ) {
1548 lineSegments.erase(nextSeg);
1549 closedContour = true;
1550 break;
1551 }
1552
1553 // otherwise, see if we can match up either end
1554 foundNextPoint = true;
1555 if(isClose(contour.front(),nextSeg->first)) {
1556 contour.push_front(nextSeg->second);
1557 }
1558 else if(isClose(contour.front(),nextSeg->second)) {
1559 contour.push_front(nextSeg->first);
1560 }
1561 else if(isClose(contour.back(),nextSeg->first)) {
1562 contour.push_back(nextSeg->second);
1563 }
1564 else if(isClose(contour.back(),nextSeg->second)) {
1565 contour.push_back(nextSeg->first);
1566 }
1567 else {
1568 foundNextPoint = false;
1569 }
1570 if(foundNextPoint) {
1571 lineSegments.erase(nextSeg);
1572 break;
1573 }
1574 }
1575 }
1576
1577 if(!closedContour) {
1578 IFCImporter::LogWarn("GetContoursInPlane3D: did not close contour");
1579 }
1580
1581 // now add the contour if we can
1582 if(contour.size() <= 2) {
1583 IFCImporter::LogWarn("GetContoursInPlane3D: discarding line/point contour");
1584 continue;
1585 }
1586 Contour c{};
1587 for(auto p : contour)
1588 {
1589 c.push_back(p);
1590 }
1591 contours.push_back(c);
1592 }
1593
1594 {
1595 std::stringstream msg;
1596 msg << "GetContoursInPlane3D: found " << contours.size() << " contours:\n";
1597
1598 for(const auto& c : contours) {
1599 msg << " Contour: \n";
1600 for(auto p : c) {
1601 msg << " " << p.x << " " << p.y << " \n";
1602 }
1603 }
1604
1605 IFCImporter::LogInfo(msg.str().c_str());
1606 }
1607
1608
1609 return contours;
1610}

1640 // working coordinate system.

1651 IfcFloat coord = -1;

1654 contour_flat.reserve(out.size());

1660 for(IfcVector3& x : out) {

1664 // (which are present, of course), this should be the same value for

1676 // but here we won't rely on this.

1680 // If this happens then the projection must have been wrong.
1681 ai_assert(vmax.Length())(void)((!!(vmax.Length())) || (Assimp::aiAssertViolation("vmax.Length()"
, "/usr/STLcheck/sekitest/assimp/code/AssetLib/IFC/IFCOpenings.cpp"
