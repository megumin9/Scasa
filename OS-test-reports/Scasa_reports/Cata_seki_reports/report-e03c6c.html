440// Figure out encumbrance penalties this clothing is involved in
441item_penalties outfit::get_item_penalties( std::list<item>::const_iterator worn_item_it,
442 const Character &c, const bodypart_id &_bp )
443{
444 std::vector<layer_level> layer = worn_item_it->get_layer();
445
446 std::vector<bodypart_id> body_parts_with_stacking_penalty;
447 std::vector<bodypart_id> body_parts_with_out_of_order_penalty;
448 std::vector<std::set<std::string>> lists_of_bad_items_within;
449
450 for( const bodypart_id &bp : c.get_all_body_parts() ) {
451 if( bp != _bp && _bp != bodypart_id( "bp_null" ) ) {
452 continue;
453 }
454 if( !worn_item_it->covers( bp ) ) {
455 continue;
456 }
457
458 std::set<std::string> bad_items_within;
459
460 // if no subparts do the old way
461 if( bp->sub_parts.empty() ) {
462 std::vector<layer_level> layer = worn_item_it->get_layer( bp );
463 bool first_integrated = true;
464 const int num_items = std::count_if( worn.begin(), worn.end(),
465 [layer, bp, &first_integrated]( const item & i ) {
466 if( i.has_layer( layer, bp ) && i.covers( bp ) && !i.has_flag( flag_SEMITANGIBLE ) ) {
467 if( i.has_flag( flag_INTEGRATED ) ) {
468 if( first_integrated ) {
469 first_integrated = false;
470 return true;
471 }
472 return false;
473 }
474 return true;
475 }
476 return false;
477 } );
478 if( num_items > 1 ) {
479 body_parts_with_stacking_penalty.push_back( bp );
480 }
481 for( auto it = worn.begin(); it != worn_item_it; ++it ) {
482 if( it->get_layer( bp ) > layer && it->covers( bp ) ) {
483 bad_items_within.insert( it->type_name() );
484 }
485 }
486 } else {
487 for( const sub_bodypart_str_id &sbp : bp->sub_parts ) {
488 if( !worn_item_it->covers( sbp ) ) {
489 continue;
490 }
491 std::vector<layer_level> layer = worn_item_it->get_layer( sbp );
492 bool first_integrated = true;
493 const int num_items = std::count_if( worn.begin(), worn.end(),
494 [layer, bp, sbp, &first_integrated]( const item & i ) {
495 if( i.has_layer( layer, sbp ) && i.covers( bp ) && !i.has_flag( flag_SEMITANGIBLE ) &&
496 i.covers( sbp ) ) {
497 if( i.has_flag( flag_INTEGRATED ) ) {
498 if( first_integrated ) {
499 first_integrated = false;
500 return true;
501 }
502 return false;
503 }
504 return true;
505 }
506 return false;
507 } );
508 if( num_items > 1 ) {
509 body_parts_with_stacking_penalty.push_back( bp );
510 }
511 for( auto it = worn.begin(); it != worn_item_it; ++it ) {
512 if( it->covers( sbp ) ) {
513 if( it->get_highest_layer( sbp ) > worn_item_it->get_highest_layer( sbp ) ) {
514 bad_items_within.insert( it->type_name() );
515 }
516 }
517 }
518 }
519 }
520
521 if( !bad_items_within.empty() ) {
522 body_parts_with_out_of_order_penalty.push_back( bp );
523 lists_of_bad_items_within.push_back( bad_items_within );
524 }
525 }
526
527 // We intersect all the lists_of_bad_items_within so that if there is one
528 // common bad item we're wearing this one over it can be mentioned in the
529 // message explaining the penalty.
530 while( lists_of_bad_items_within.size() > 1 ) {
531 std::set<std::string> intersection_of_first_two;
532 std::set_intersection(
533 lists_of_bad_items_within[0].begin(), lists_of_bad_items_within[0].end(),
534 lists_of_bad_items_within[1].begin(), lists_of_bad_items_within[1].end(),
535 std::inserter( intersection_of_first_two, intersection_of_first_two.begin() )
536 );
537 lists_of_bad_items_within.erase( lists_of_bad_items_within.begin() );
538 lists_of_bad_items_within[0] = std::move( intersection_of_first_two );
539 }
540
541 if( lists_of_bad_items_within.empty() ) {
542 lists_of_bad_items_within.emplace_back();
543 }
544
545 return { std::move( body_parts_with_stacking_penalty ),
546 std::move( body_parts_with_out_of_order_penalty ),
547 std::move( lists_of_bad_items_within[0] ) };
548}
