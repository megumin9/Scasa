1928bool worldfactory::valid_worldname( const std::string &name, bool automated ) const
1929{
1930 std::string msg;
1931
1932 if( name.empty() ) {
1933 msg = _( "World name cannot be empty!" )( ( []( const auto & arg ) { static auto cache = detail::

translation_argument_identity( "World name cannot be empty!"
) ) );
1934 } else if( name == "save" || name == "TUTORIAL" || name == "DEFENSE" ) {
1935 msg = string_format( _( "%s is a reserved name!" )( ( []( const auto & arg ) { static auto cache = detail::
get_local_translation_cache( arg ); return cache( arg ); } )(
translation_argument_identity( "%s is a reserved name!" ) ) ), name );
1936 } else if( has_world( name ) ) {
1937 msg = string_format( _( "A world named %s already exists!" )( ( []( const auto & arg ) { static auto cache = detail::
get_local_translation_cache( arg ); return cache( arg ); } )(
translation_argument_identity( "A world named %s already exists!"
) ) ), name );
1938 } else {
1939 // just check the raw bytes because unicode characters are always acceptable
1940 bool allowed = true;
1941 for( const char ch : name ) {
1942 // Convert to unsigned char because `std::isprint` is undefined for
1943 // values unrepresentable by unsigned char which is not EOF.
1944 const unsigned char uc = static_cast<unsigned char>( ch );
1945 if( !is_char_allowed( uc ) ) {
1946 if( std::isprint( uc ) ) {
1947 msg = string_format( _( "World name contains invalid character: '%c'" )( ( []( const auto & arg ) { static auto cache = detail::
get_local_translation_cache( arg ); return cache( arg ); } )(
translation_argument_identity( "World name contains invalid character: '%c'"
) ) ), uc );
1948 } else {
1949 msg = string_format( _( "World name contains invalid character: 0x%x" )( ( []( const auto & arg ) { static auto cache = detail::
get_local_translation_cache( arg ); return cache( arg ); } )(
translation_argument_identity( "World name contains invalid character: 0x%x"
) ) ), uc );
1950 }

1957 }

2086void worldfactory::delete_world( const std::string &worldname, const bool delete_folder )
2087{
2088 std::string worldpath = get_world( worldname )->folder_path();
2089 std::set<std::string> directory_paths;
2090
2091 auto file_paths = get_files_from_path( "", worldpath, true, true );
2092 if( !delete_folder ) {
2093 std::vector<std::string>::iterator forbidden = find_if( file_paths.begin(), file_paths.end(),
2094 isForbidden );
2095 while( forbidden != file_paths.end() ) {
2096 file_paths.erase( forbidden );
2097 forbidden = find_if( file_paths.begin(), file_paths.end(), isForbidden );
2098 }
2099 }
2100 for( auto &file_path : file_paths ) {
2101 // strip to path and remove worldpath from it
2102 std::string part = file_path.substr( worldpath.size(),
2103 file_path.find_last_of( "/\\" ) - worldpath.size() );
2104 size_t last_separator = part.find_last_of( "/\\" );
2105 while( last_separator != std::string::npos && part.size() > 1 ) {
2106 directory_paths.insert( part );
2107 part = part.substr( 0, last_separator );
2108 last_separator = part.find_last_of( "/\\" );
2109 }
2110 }
2111
2112 for( auto &file : file_paths ) {
2113 remove_file( file );
2114 }
2115 // Trying to remove a non-empty parent directory before a child
2116 // directory will fail. Removing directories in reverse order
2117 // will prevent this situation from arising.
2118 for( auto it = directory_paths.rbegin(); it != directory_paths.rend(); ++it ) {
2119 remove_directory( worldpath + *it );
2120 }
2121 if( delete_folder ) {
2122 remove_directory( worldpath );
2123 remove_world( worldname );
2124 } else {
2125 get_world( worldname )->world_saves.clear();
2126 }
