676 PICKUP_RANGE );

680 // Time (roughly) in turns to cut up the corpse
681 case creature_size::tiny:
682 time_to_cut = 150;
683 break;
684 case creature_size::small:
685 time_to_cut = 300;
686 break;
687 case creature_size::medium:
688 time_to_cut = 450;
689 break;
690 case creature_size::large:
691 time_to_cut = 600;

698 time_to_cut = 450; // default to medium
699 break;
700 }
701
702 // At factor 0, base 100 time_to_cut remains 100. At factor 50, it's 50 , at factor 75 it's 25
703 time_to_cut *= std::max( 25, 100 - factor );
704
705 switch( action ) {
706 case butcher_type::QUICK:
707 case butcher_type::BLEED:
708 break;
709 case butcher_type::FULL:
710 if( !corpse_item.has_flag( flag_FIELD_DRESS ) || corpse_item.has_flag( flag_FIELD_DRESS_FAILED ) ) {
711 time_to_cut *= 6;
712 } else {
713 time_to_cut *= 4;
714 }
715 break;
716 case butcher_type::FIELD_DRESS:
717 case butcher_type::SKIN:
718 time_to_cut *= 2;
719 break;
720 case butcher_type::QUARTER:
721 time_to_cut /= 4;
722 if( time_to_cut < 1200 ) {
723 time_to_cut = 1200;
724 }
725 break;
726 case butcher_type::DISMEMBER:
727 time_to_cut /= 10;
728 if( time_to_cut < 600 ) {
729 time_to_cut = 600;
730 }
731 break;
732 case butcher_type::DISSECT:
733 time_to_cut *= 6;
734 break;
735 case butcher_type::NUM_TYPES:
736 debugmsg( "ERROR: Invalid butcher_type" )realDebugmsg("/usr/STLcheck/sekitest/Cataclysm-DDA/src/activity_handlers.cpp"
, "736", __PRETTY_FUNCTION__, "ERROR: Invalid butcher_type");
737 break;
738 }
739
740 if( corpse_item.has_flag( flag_QUARTERED ) ) {
741 time_to_cut /= 4;
742 }
743 time_to_cut *= ( 1.0f - ( get_player_character().get_num_crafting_helpers( 3 ) / 10.0f ) );
744 return time_to_cut;
745}

747// this function modifies the input weight by its damage level, depending on the bodypart
748static int corpse_damage_effect( int weight, const harvest_drop_type_id &entry_type,
749 int damage_level )
750{
751 const float slight_damage = 0.9f;
752 const float damage = 0.75f;
753 const float high_damage = 0.5f;
754 const int destroyed = 0;
755
756 switch( damage_level ) {
757 case 2:
758 // "damaged"
759 if( entry_type == harvest_drop_offal ) {
760 return std::round( weight * damage );
761 }
762 if( entry_type == harvest_drop_skin ) {
763 return std::round( weight * damage );
764 }
765 if( entry_type == harvest_drop_flesh ) {
766 return std::round( weight * slight_damage );
767 }
768 break;
769 case 3:
770 // "mangled"
771 if( entry_type == harvest_drop_offal ) {
772 return destroyed;
773 }
774 if( entry_type == harvest_drop_skin ) {
775 return std::round( weight * high_damage );
776 }
777 if( entry_type == harvest_drop_bone ) {
778 return std::round( weight * slight_damage );
779 }
780 if( entry_type == harvest_drop_flesh ) {
781 return std::round( weight * damage );
782 }
783 break;
784 case 4:
785 // "pulped"
786 if( entry_type == harvest_drop_offal ) {
787 return destroyed;
788 }
789 if( entry_type == harvest_drop_skin ) {
790 return destroyed;
791 }
792 if( entry_type == harvest_drop_bone ) {
793 return std::round( weight * damage );
794 }
795 if( entry_type == harvest_drop_flesh ) {
796 return std::round( weight * high_damage );
797 }
798 break;
799 default:
800 // "bruised" modifier is almost impossible to avoid; also includes no modifier (zero damage)
801 break;
802 }
803 return weight;
804}

822static int roll_butchery_dissect( int skill_level, int dex, int tool_quality )
823{
824 double skill_shift = 0.0;
825 ///\EFFECT_SURVIVAL randomly increases butcher rolls
826 skill_shift += rng_float( 0, skill_level - 3 );
827 ///\EFFECT_DEX >8 randomly increases butcher rolls, slightly, <8 decreases
828 skill_shift += rng_float( 0, dex - 8 ) / 4.0;
829
830 if( tool_quality < 0 ) {
831 skill_shift -= rng_float( 0, -tool_quality / 5.0 );
832 } else {
833 skill_shift += std::min( tool_quality, 4 );
834 }
835
836 return static_cast<int>( std::round( skill_shift ) );
837}

839// A number between 0 and 1 that represents how much usable material can be harvested
840// as a fraction of the maximum possible amount.
841static double butchery_dissect_yield_mult( int skill_level, int dex, int tool_quality )
842{
843 const double skill_score = skill_level / 10.0;
844 const double tool_score = ( tool_quality + 50.0 ) / 100.0;
845 const double dex_score = dex / 20.0;
846 return 0.5 * clamp( skill_score, 0.0, 1.0 ) +
847 0.3 * clamp( tool_score, 0.0, 1.0 ) +
848 0.2 * clamp( dex_score, 0.0, 1.0 );
849}

1562 liquid.type_name( 1 ) );

1566 here.furn_set( source_pos, f_fvat_wood_empty );

1575 case liquid_source_type::INFINITE_MAP:

1589 }
1590
1591 } catch( const std::runtime_error &err ) {
1592 debugmsg( "error in activity data: \"%s\"", err.what() )realDebugmsg("/usr/STLcheck/sekitest/Cataclysm-DDA/src/activity_handlers.cpp"

1774 weap_stab = weapon->damage_melee( damage_stab );

1777 mess_radius = 2;
1778 }

1782 const int cut_power = std::max( weap_cut, weap_stab / 2 );

1787

1789 pulp_power *= 40 + you->get_skill_level( skill_survival ) * 5;
1790

1792 // use this to collect how many corpse are pulped
1793 int &num_corpses = act->index;
1794 map_stack corpse_pile = here.i_at( pos );
1795 for( item &corpse : corpse_pile ) {
1796 const mtype *corpse_mtype = corpse.get_mtype();
1797 const bool acid_immune = you->is_immune_damage( damage_acid ) ||
1798 you->is_immune_field( fd_acid );

1800 ( ( std::find( act->str_values.begin(), act->str_values.end(), "auto_pulp_no_acid" ) !=
1801 act->str_values.end() && corpse_mtype->bloodType().obj().has_acid ) && !acid_immune ) ) {

1803 continue;
