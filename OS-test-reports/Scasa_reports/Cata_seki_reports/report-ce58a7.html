634static void field_processor_fd_electricity( const tripoint &p, field_entry &cur,
635 field_proc_data &pd )
636{
637 // Higher chance of spreading for intense fields
638 int current_intensity = cur.get_field_intensity();
639 if( one_in( current_intensity * 2 ) ) {
640 return;
641 }
642
643 const int spread_intensity_cap = 3 + std::max( ( current_intensity - 3 ) / 2, 0 );
644
645 std::vector<tripoint> grounded_tiles;
646 std::vector<tripoint> tiles_with_creatures;
647 std::vector<tripoint> other_tiles;
648
649 bool valid_candidates = false;
650 for( const tripoint &dst : points_in_radius( p, 1 ) ) {
651 // Skip tiles with intense fields
652 const field_type_str_id &field_type = pd.here.get_applicable_electricity_field( dst );
653 if( field_entry *field = pd.here.get_field( dst, field_type ) ) {
654 if( field->get_field_intensity() >= spread_intensity_cap ) {
655 continue;
656 }
657 }
658
659 if( pd.here.impassable( dst ) ) {
660 grounded_tiles.push_back( dst );
661 } else {
662 if( get_creature_tracker().creature_at( dst ) ) {
663 tiles_with_creatures.push_back( dst );
664 } else {
665 other_tiles.push_back( dst );
666 }
667 }
668 valid_candidates = true;
669 }
670
671 if( !valid_candidates ) {
672 return;
673 }
674
675 const bool here_impassable = pd.here.impassable( p );
676 int grounded_weight = here_impassable ? 1 : 6;
677 int creature_weight = here_impassable ? 5 : 6;
678 int other_weight = here_impassable ? 0 : 1;
679
680 std::vector<tripoint> *target_vector = nullptr;
681 while( current_intensity > 0 ) {
682
683 if( here_impassable && one_in( 3 ) ) {
684 // Electricity in impassable tiles will find a way to the ground sometimes
685 cur.set_field_intensity( --current_intensity );
686 continue;
687 }
688
689 const int vector_choice = bucket_index_from_weight_list( std::vector<int>( {
690 grounded_tiles.empty() ? 0 : grounded_weight,
691 tiles_with_creatures.empty() ? 0 : creature_weight,
692 other_tiles.empty() ? 0 : other_weight
693 } ) );
694
695 switch( vector_choice ) {
696 default:
697 case 0:
698 if( here_impassable && !one_in( 5 ) ) {
699 return;
700 }
701 target_vector = &grounded_tiles;
702 break;
703 case 1:
704 target_vector = &tiles_with_creatures;
705 break;
706 case 2:
707 target_vector = &other_tiles;
708 break;
709 }
710
711 if( target_vector->empty() ) {
712 return;
713 }
714
715 int vector_index = rng( 0, target_vector->size() - 1 );
716 auto target_it = target_vector->begin() + vector_index;
717 tripoint target_point = *target_it;
718
719 const field_type_str_id &field_type = pd.here.get_applicable_electricity_field( target_point );
720
721 // Intensify target field if it exists, create a new one otherwise
722 if( field_entry *target_field = pd.here.get_field( target_point, field_type ) ) {
723 int target_field_intensity = target_field->get_field_intensity();
724 target_field->set_field_intensity( ++target_field_intensity );
725 if( target_field_intensity >= spread_intensity_cap ) {
726 target_vector->erase( target_it );
727 }
728 } else {
729 pd.here.add_field( target_point, field_type, 1, cur.get_field_age() + 1_turns );
730 }
731
732 cur.set_field_intensity( --current_intensity );
733
734 if( one_in( current_intensity * 2 ) ) {
735 // Weaker fields have a harder time spreading
736 break;
737 }
738 }
739}

918void field_processor_fd_fire( const tripoint &p, field_entry &cur, field_proc_data &pd )
919{
920 const field_type_id fd_fire = ::fd_fire;
921 map &here = pd.here;
922 maptile &map_tile = pd.map_tile;
923 const oter_id om_ter = pd.om_ter;
924
925 cur.set_field_age( std::max( -24_hours, cur.get_field_age() ) );
926 // Entire objects for ter/frn for flags
927 bool sheltered = g->is_sheltered( p );
928 weather_manager &weather = get_weather();
929 int winddirection = weather.winddirection;
930 int windpower = get_local_windpower( weather.windspeed, om_ter, tripoint_abs_ms( p ), winddirection,
931 sheltered );
932 const ter_t &ter = map_tile.get_ter_t();
933 const furn_t &frn = map_tile.get_furn_t();
934
935 // We've got ter/furn cached, so let's use that
936 const bool is_sealed = ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_SEALED ) &&
937 !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_ALLOW_FIELD_EFFECT );
938 // Smoke generation probability, consumed items count
939 int smoke = 0;
940 int consumed = 0;
941 // How much time to add to the fire's life due to burned items/terrain/furniture
942 time_duration time_added = 0_turns;
943 // Checks if the fire can spread
944 const bool can_spread = !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FIRE_CONTAINER );
945 // If the flames are in furniture with fire_container flag like brazier or oven,
946 // they're fully contained, so skip consuming terrain
947 const bool can_burn = ( ter.is_flammable() || frn.is_flammable() ) &&
948 !ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FIRE_CONTAINER );
949 // The huge indent below should probably be somehow moved away from here
950 // without forcing the function to use i_at( p ) for fires without items
951 if( !is_sealed && map_tile.get_item_count() > 0 ) {
952 map_stack items_here = here.i_at( p );
953 std::vector<item> new_content;
954 for( auto it = items_here.begin(); it != items_here.end(); ) {
955 if( it->will_explode_in_fire() ) {
956 // We need to make a copy because the iterator validity is not predictable
957 item copy = *it;
958 it = items_here.erase( it );
959 if( copy.detonate( p, new_content ) ) {
960 // Need to restart, iterators may not be valid
961 it = items_here.begin();
962 }
963 } else {
964 ++it;
965 }
966 }
967
968 fire_data frd( cur.get_field_intensity(), !can_spread );
969 // The highest # of items this fire can remove in one turn
970 int max_consume = cur.get_field_intensity() * 2;
971
972 for( auto fuel = items_here.begin(); fuel != items_here.end() && consumed < max_consume; ) {
973 // `item::burn` modifies the charges in order to simulate some of them getting
974 // destroyed by the fire, this changes the item weight, but may not actually
975 // destroy it. We need to spawn products anyway.
976 const units::mass old_weight = fuel->weight( false );
977 bool destroyed = fuel->burn( frd );
978 // If the item is considered destroyed, it may have negative charge count,
979 // see `item::burn?. This in turn means `item::weight` returns a negative value,
980 // which we can not use, so only call `weight` when it's still an existing item.
981 const units::mass new_weight = destroyed ? 0_gram : fuel->weight( false );
982 if( old_weight != new_weight ) {
983 here.create_burnproducts( p, *fuel, old_weight - new_weight );
984 }
985
986 if( destroyed ) {
987 // If we decided the item was destroyed by fire, remove it.
988 // But remember its contents, except for irremovable mods, if any
989 const std::list<item *> content_list = fuel->all_items_top();
990 for( item *it : content_list ) {
991 if( !it->is_irremovable() ) {
992 new_content.emplace_back( *it );
993 }
994 }
995 fuel = items_here.erase( fuel );
996 consumed++;
997 } else {
998 ++fuel;
999 }
1000 }
1001
1002 here.spawn_items( p, new_content );
1003 smoke = roll_remainder( frd.smoke_produced );
1004 time_added = 1_turns * roll_remainder( frd.fuel_produced );
1005 }
1006
1007 int part;
1008 // Get the part of the vehicle in the fire (_internal skips the boundary check)
1009 vehicle *veh = here.veh_at_internal( p, part );
1010 if( veh != nullptr ) {
1011 veh->damage( here, part, cur.get_field_intensity() * 10, damage_heat, true );
1012 // Damage the vehicle in the fire.
1013 }
1014 if( can_burn ) {
1015 if( ter.has_flag( ter_furn_flag::TFLAG_SWIMMABLE ) ) {
1016 // Flames die quickly on water
1017 cur.set_field_age( cur.get_field_age() + 4_minutes );
1018 }
1019
1020 // Consume the terrain we're on
1021 if( ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FLAMMABLE ) ) {
1022 // The fire feeds on the ground itself until max intensity.
1023 time_added += 1_turns * ( 5 - cur.get_field_intensity() );
1024 smoke += 2;
1025 smoke += static_cast<int>( windpower / 5 );
1026 if( cur.get_field_intensity() > 1 &&
1027 one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1028 here.destroy( p, false );
1029 }
1030
1031 } else if( ter_furn_has_flag( ter, frn, ter_furn_flag::TFLAG_FLAMMABLE_HARD ) &&
1032 one_in( 3 ) ) {
1033 // The fire feeds on the ground itself until max intensity.
1034 time_added += 1_turns * ( 4 - cur.get_field_intensity() );
1035 smoke += 2;
1036 smoke += static_cast<int>( windpower / 5 );
1037 if( cur.get_field_intensity() > 1 &&
1038 one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1039 here.destroy( p, false );
1040 }
1041
1042 } else if( ter.has_flag( ter_furn_flag::TFLAG_FLAMMABLE_ASH ) ) {
1043 // The fire feeds on the ground itself until max intensity.
1044 time_added += 1_turns * ( 5 - cur.get_field_intensity() );
1045 smoke += 2;
1046 smoke += static_cast<int>( windpower / 5 );
1047 if( cur.get_field_intensity() > 1 &&
1048 one_in( 200 - cur.get_field_intensity() * 50 ) ) {
1049 here.spawn_item( p, "ash", 1, rng( 10, 1000 ) );
1050 if( p.z > 0 ) {
1051 // We're in the air. Need to invalidate the furniture otherwise it'll cause problems
1052 here.furn_set( p, f_null );
1053 here.ter_set( p, t_open_air );
1054 } else if( p.z < -1 ) {
1055 // We're deep underground, in bedrock. Whatever terrain was here is burned to the ground, leaving only the carved out rock (including ceiling)
1056 here.ter_set( p, t_rock_floor );
1057 } else {
1058 // Need to invalidate the furniture otherwise it'll cause problems when supporting terrain collapses
1059 here.furn_set( p, f_null );
1060 here.ter_set( p, t_dirt );
1061 }
