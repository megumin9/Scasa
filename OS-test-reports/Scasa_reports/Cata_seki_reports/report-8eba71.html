599{
600 //~ translators: place some random 1-width characters here in your language if possible, or leave it as is

608 // a trailing NULL terminator is necessary for utf8_width function
609 // NOLINTNEXTLINE(modernize-avoid-c-arrays)
610 char transformation[2] = { 0 };
611 for( size_t i = 0; i < w_str.size(); ++i ) {
612 transformation[0] = f();
613 std::string this_char = wstr_to_utf8( std::wstring( 1, w_str[i] ) );
614 // mk_wcwidth, which is used by utf8_width, might return -1 for some values, such as newlines 0x0A

, "628", __PRETTY_FUNCTION__, "target character isn't narrow"
);
629 }
630 // A 2-width wide character in the original string should be replace by two narrow characters
631 w_str.replace( i, 1, utf8_to_wstr( std::string( utf8_width( this_char ), transformation[0] ) ) );
632 }

764holiday get_holiday_from_time( std::time_t time, bool force_refresh )
765{
766 static holiday cached_holiday = holiday::none;
767 static bool is_cached = false;
768
769 if( force_refresh ) {
770 is_cached = false;
771 }
772 if( is_cached ) {
773 return cached_holiday;
774 }
775
776 is_cached = true;
777
778 bool success = false;
779
780 std::tm local_time;
781 std::time_t current_time = time == 0 ? std::time( nullptr ) : time;
782
783 /* necessary to pass LGTM, as threadsafe version of localtime differs by platform */
784#if defined(_WIN32)
785
786 errno_t err = localtime_s( &local_time, &current_time );
787 if( err == 0 ) {
788 success = true;
789 }
790
791#else
792
793 success = !!localtime_r( &current_time, &local_time );
794
795#endif
796
797 if( success ) {
798
799 const int month = local_time.tm_mon + 1;
800 const int day = local_time.tm_mday;
801 const int wday = local_time.tm_wday;
802 const int year = local_time.tm_year + 1900;
803
804 /* check date against holidays */
805 if( month == 1 && day == 1 ) {
806 cached_holiday = holiday::new_year;
807 return cached_holiday;
808 }
809 // only run easter date calculation if currently March or April
810 else if( ( month == 3 || month == 4 ) && is_easter( day, month, year ) ) {
811 cached_holiday = holiday::easter;
812 return cached_holiday;
813 } else if( month == 7 && day == 4 ) {
814 cached_holiday = holiday::independence_day;
815 return cached_holiday;
816 }
817 // 13 days seems appropriate for Halloween
818 else if( month == 10 && day >= 19 ) {
819 cached_holiday = holiday::halloween;
820 return cached_holiday;
821 } else if( month == 11 && ( day >= 22 && day <= 28 ) && wday == 4 ) {
822 cached_holiday = holiday::thanksgiving;
823 return cached_holiday;
824 }
825 // For the 12 days of Christmas, my true love gave to me...
826 else if( month == 12 && ( day >= 14 && day <= 25 ) ) {
827 cached_holiday = holiday::christmas;
828 return cached_holiday;
829 }
830 }
831 // fall through to here if localtime fails, or none of the day tests hit
832 cached_holiday = holiday::none;
833 return cached_holiday;
834}

836int bucket_index_from_weight_list( const std::vector<int> &weights )
837{
838 int total_weight = std::accumulate( weights.begin(), weights.end(), int( 0 ) );
839 if( total_weight < 1 ) {
840 return 0;
841 }
842 const int roll = rng( 0, total_weight - 1 );
843 int index = 0;
844 int accum = 0;
845 for( int w : weights ) {
846 accum += w;
847 if( accum > roll ) {
848 break;
849 }
850 index++;
851 }
852 return index;
853}
